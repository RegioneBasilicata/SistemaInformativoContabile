function accertamenti (p_anno in NUMBER, p_capitolo in varchar2) return number IS
tot_accertamenti number;
begin
select nvl(sum(document_amount),0) into tot_accertamenti
from fin_t_documenti
where doc_type ='ACC'
and segment8 = p_anno
and nvl(deletion_status_flag,'N') = 'N'
and segment2 = p_capitolo;

return(tot_accertamenti);
end;
