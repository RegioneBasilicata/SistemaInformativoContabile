function analizzo_ore(w_entrata in NUMBER, 
                                         w_uscita in NUMBER, 
                                         w_entrata_turno in NUMBER,
                                         w_uscita_turno in NUMBER) return number IS
w_ore number := 0; 
BEGIN
-- HO 4 CASI POSSIBILI:
-- 1 ENTRATA MINORE-UGUALE ENTRATA TURNO E USCITA MINORE-UGUALE USCITA TURNO  
IF w_entrata <= w_entrata_turno and
    w_uscita > w_entrata_turno and
    w_uscita <= w_uscita_turno
  THEN
   w_ore:= w_uscita - w_entrata_turno;  
-- 2 ENTRATA MINORE-UGUALE ENTRATA TURNO E USCITA MAGGIORE USCITA TURNO  IF 
 ELSIF w_entrata <= w_entrata_turno and
    w_uscita > w_uscita_turno
    THEN w_ore:= w_uscita_turno - w_entrata_turno;  
-- 3 ENTRATA MAGGIORE ENTRATA TURNO E USCITA MINORE-UGUALE USCITA TURNO  
   ELSIF w_entrata > w_entrata_turno and
         w_uscita <= w_uscita_turno
     THEN w_ore:= w_uscita - w_entrata;  
-- 4 ENTRATA MAGGIORE ENTRATA TURNO E MINOREUGUALE USCITA TURNO E USCITA MAGGIORE USCITA TURNO  
     ELSIF w_entrata > w_entrata_turno and
           w_entrata <= w_uscita_turno and
           w_uscita > w_uscita_turno
      THEN w_ore:= w_uscita_turno - w_entrata;  
END IF;
   return w_ore;
END;
