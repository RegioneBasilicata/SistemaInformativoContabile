FUNCTION      assestamento_fpvcapitale_pl1(p_anno in NUMBER) return number IS
importo number(15,2) := 0; 
vdata_atto_assestamento date;

BEGIN
 begin
 select data_atto_assestamento 
 into vdata_atto_assestamento
 from fin_t_dati_generali 
 where anno_finanziario=p_anno - 1;
 exception when no_data_found then null;
 end;
 select 
 nvl(sum(decode(substr(d.doc_type,7,3),'NEG',(d.document_amount * -1),d.document_amount)),0) 
 into importo
 from  fin_t_documenti d, 
      fin_t_articoli a
 where d.segment8=p_anno 
 and substr(d.segment2,1,1) = 'U'
 and d.segment9 in ('FPV','FPV_REG')
 and 
 (
  d.doc_type in 
 ('A_PL1_POS','A_PL1_NEG') 
 or (doc_type in ('V_PL1_POS','V_PL1_NEG') and nvl(attribute10,'N')<>'Y') and trunc(to_date(attribute17,'dd/mm/yyyy'),'dd') < trunc(nvl(vdata_atto_assestamento,to_date(attribute17,'dd/mm/yyyy')+1),'dd') )
 and d.segment8 = a.anno 
 and d.segment2 = a.eu||a.codice 
 and a.nb_titolo = '2'; 
 
 return nvl(importo,0);
END;
