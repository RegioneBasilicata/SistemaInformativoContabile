function AVANZO (p_anno number) return number is
residui_passivi number := 0;
residui_attivi number := 0;
avanzo_calcolato number := 0;

begin 
    select nvl(sum(residuo_imp_data(d.doc_id,to_date('3112'||p_anno,'ddmmyyyy'))),0) into residui_attivi
    from fin_t_documenti d
    where d.doc_type = 'ACC'
    and nvl(d.deletion_status_flag,'N') = 'N'
    and d.segment8 <= p_anno
    and residuo_imp_data(d.doc_id,to_date('3112'||p_anno,'ddmmyyyy')) > 0;
    
    
    select nvl(sum(residuo_imp_data(d.doc_id,to_date('3112'||p_anno,'ddmmyyyy'))),0) into residui_passivi
    from fin_t_documenti d
    where d.doc_type like 'IMP%'
    and nvl(d.deletion_status_flag,'N') = 'N'
    and segment8 <= p_anno
    and residuo_imp_data(d.doc_id,to_date('3112'||p_anno,'ddmmyyyy')) > 0;
    
    avanzo_calcolato := fondo_cassa(p_anno) + residui_attivi - residui_passivi;
    return(avanzo_calcolato);
END;
