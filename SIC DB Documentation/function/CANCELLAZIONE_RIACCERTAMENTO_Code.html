function cancellazione_riaccertamento (p_anno in number, p_capitolo in varchar2) return number is

importo  number(15,2) := 0;

begin

select sum(importo_riaccertamento) into importo
from fin_T_riaccertamenti
where anno_bilancio_riaccertamento=p_anno+1
and capitolo=p_capitolo
and nvl(flag_elimina,'N')='N'--solo documenti riaccertati quindi solo N Luigi Pierro 05-06-2019
and tipo_documento like 'IMP%';
return nvl(importo,0);


end;
