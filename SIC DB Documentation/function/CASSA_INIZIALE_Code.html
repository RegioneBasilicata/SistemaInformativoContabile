function      cassa_iniziale(p_anno in NUMBER, p_capitolo in varchar2) return number IS
importo number(15,2) := 0; 
vdata_approvazione_previsione date;


BEGIN

 begin
 select data_approvazione_previsione 
 into vdata_approvazione_previsione
 from fin_t_dati_generali 
 where anno_finanziario=p_anno;
 exception when no_data_found then null;
 end;
 
 
 select 
 nvl(sum(decode(substr(d.doc_type,7,3),'NEG',(d.document_amount * -1),d.document_amount)),0) 
 into importo
 from  fin_t_documenti d
 where d.segment8 = p_anno
 and d.segment2 = p_capitolo
 and 
 (
 d.doc_type in 
 ('B_CAS_POS'
 ,'B_CAS_NEG'
 ) or (doc_type in ('V_CAS_POS','V_CAS_NEG') and trunc(date_created,'dd') < trunc(nvl(vdata_approvazione_previsione,date_created+1),'dd')));
 
--or determinato per il bilancio provvisorio in quanto in fase di approvazione del bilancio stanno facendo delle variazioni.

 return nvl(importo,0);
END;
