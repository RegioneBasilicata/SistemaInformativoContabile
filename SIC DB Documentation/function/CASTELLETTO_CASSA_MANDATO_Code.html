function castelletto_cassa_mandato(p_anno in number, p_capitolo in varchar2, p_mandato in number) return number IS

importo number(15,2) := 0; 
cassa number(15,2):=0;
mandati_emessi number(15,2):=0;
data_mandato number;
anno0 number(4);
bilancio_provvisorio varchar2(1);
anno_appoggio number(4);

BEGIN

begin
  select 
    anno_provvisiorio1
   ,bilancio_previsione_aperto
   ,to_number(to_char(sysdate,'yyyy'))
  into 
   anno0
  ,bilancio_provvisorio
  ,anno_appoggio
  from fin_t_dati_generali2;
  exception when no_data_found then 
    bilancio_provvisorio := 'N';
    anno0 := anno_appoggio;
 end;


--calcolo la disponibilità di cassa.
if bilancio_provvisorio = 'S' and anno0 = p_anno -- provvisorio attivo
   then 
   select pluriennale1_definitivo(p_anno,p_capitolo) + residuo_iniziale(p_anno,p_capitolo)
   into cassa
   from dual;
   else
   select cassa_definitiva(p_anno,p_capitolo) - previsione_da_nascondere(p_anno,p_capitolo,'S') - assestamento_da_nascondere(p_anno,p_capitolo,'S') 
   into cassa
   from dual;
end if;


-- recupero la data del mandato al fine di escludere gli storni emessi successivamente a quella data.
select to_number(to_char(date_created,'yyyymmdd')) 
into data_mandato
from fin_t_documenti
where doc_type = 'MAND'
and doc_sequence_value=p_mandato
and segment8=p_anno
and segment2=p_capitolo;
       
--sommo tutti i mandati emessi per anno, capitolo e numerazione <= numero del mandato, escludendo i mandati con storni emessi in data successiva alla data del mandato di riferimento.
select sum(nvl(a.document_amount,0)) 
into mandati_emessi
from fin_t_documenti a
where a.doc_type = 'MAND'
and a.segment8=p_anno
and a.segment2=p_capitolo
and a.doc_sequence_value<= p_mandato
and a.doc_id not in (select b.previous_doc_id from fin_t_documenti b where b.doc_type = 'ST-MAND' and b.segment8=p_anno and b.segment2= p_capitolo and to_number(to_char(b.date_created,'yyyymmdd'))<=data_mandato);

--viene ritornata la disponibilità di cassa rimanente. Il mandato di riferimento è già sottratto a questo valore (vedi condizione select precedente : and a.doc_sequence_value<= p_mandato)
importo := nvl(cassa,0) - nvl(mandati_emessi,0);

return nvl(importo,0);

END;
