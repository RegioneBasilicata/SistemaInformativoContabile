function castelletto_cassa_reversali(p_anno in number, p_capitolo in varchar2, p_reversale in number) return number IS

importo number(15,2) := 0; 
cassa number(15,2):=0;
reversali_emesse number(15,2):=0;
data_reversale number;
anno0 number(4);
bilancio_provvisorio varchar2(1);
anno_appoggio number(4);


BEGIN

begin
  select 
    anno_provvisiorio1
   ,bilancio_previsione_aperto
   ,to_number(to_char(sysdate,'yyyy'))
  into 
   anno0
  ,bilancio_provvisorio
  ,anno_appoggio
  from fin_t_dati_generali2;
  exception when no_data_found then 
    bilancio_provvisorio := 'N';
    anno0 := anno_appoggio;
 end;

--calcolo la disponibilità di cassa.
if bilancio_provvisorio = 'S' and anno0 = p_anno -- provvisorio attivo
   then 
   select pluriennale1_definitivo(p_anno,p_capitolo) + residuo_iniziale(p_anno,p_capitolo)
   into cassa
   from dual;
   else
   select cassa_definitiva(p_anno,p_capitolo) - previsione_da_nascondere(p_anno,p_capitolo,'S') - assestamento_da_nascondere(p_anno,p_capitolo,'S') 
   into cassa
   from dual;
end if;

-- recupero la data della reversale al fine di escludere gli storni emessi successivamente a quella data.
select to_number(to_char(date_created,'yyyymmdd')) 
into data_reversale
from fin_t_documenti
where doc_type = 'REV'
and doc_sequence_value=p_reversale
and segment8=p_anno
and segment2=p_capitolo;
       
--sommo tutte le reversali emessi per anno, capitolo e numerazione <= numero del mandato, escludendo le reversali con storni emessi in data successiva alla data dalla reversale di riferimento.
select sum(nvl(a.document_amount,0)) 
into reversali_emesse
from fin_t_documenti a
where a.doc_type = 'REV'
and a.segment8=p_anno
and a.segment2=p_capitolo
and a.doc_sequence_value<= p_reversale
and a.doc_id not in (select b.previous_doc_id from fin_t_documenti b where b.doc_type = 'ST-REV' and b.segment8=p_anno and b.segment2= p_capitolo and to_number(to_char(b.date_created,'yyyymmdd'))<=data_reversale);

--viene ritornata la disponibilità di cassa rimanente. La reversale di riferimento è già sottratta a questo valore (vedi condizione select precedente : and a.doc_sequence_value<= p_reversale)
importo := nvl(cassa,0) - nvl(reversali_emesse,0);

return nvl(importo,0);

END;
