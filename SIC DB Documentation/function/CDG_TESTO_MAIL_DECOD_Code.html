TEXT 
FUNCTION cdg_testo_mail_decod 
(p_cod_tabtesti IN NUMBER,
 p1 IN VARCHAR2, p2 IN VARCHAR2, p3 IN VARCHAR2, p4 IN VARCHAR2, p5 IN VARCHAR2,
 p6 IN VARCHAR2, p7 IN VARCHAR2, p8 IN VARCHAR2, p9 IN VARCHAR2, p10 IN VARCHAR2,
 p11 IN VARCHAR2, p12 IN VARCHAR2, p13 IN VARCHAR2, p14 IN VARCHAR2, p15 IN VARCHAR2) 
RETURN VARCHAR2 IS

 v VARCHAR2(4000);
BEGIN

  SELECT testo 
  INTO v
  FROM fin_t_cdg_testi_mail
  WHERE codice = p_cod_tabtesti;
  v := REPLACE(v,'<desc_str_richiedente>',p1);
  v := REPLACE(v,'<cod_str_richiedente>',p2);
  v := REPLACE(v,'<desc_str_titolare>',p3);
  v := REPLACE(v,'<cod_str_titolare>',p4);
  v := REPLACE(v,'<capitolo_rich>',p5);
  v := REPLACE(v,'<progr_richiesta>',p6);
  v := REPLACE(v,'<importo_rich>',p7);  
  v := REPLACE(v,'<nominativo_operatore>',p8);
  v := REPLACE(v,'<data_richiesta>',p9);
  v := REPLACE(v,'<cod_str_tit_cdg>',p10);
  v := REPLACE(v,'<desc_str_competente>',p11);
  v := REPLACE(v,'<cod_str_competente>',p12);
  v := REPLACE(v,'<importo_tit>',p13);
  v := REPLACE(v,'<importo_cdg>',p14);
  v := REPLACE(v,'<anno_finanziario>',p15);

  RETURN v;
 
END;
