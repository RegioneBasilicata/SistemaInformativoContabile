function CodiceAutoparco return number as 
codice number; 
FoglioAutoparco number;
i number:=0;
j number:=0;
cursor NumeroAutoparco is 
select distinct num_foglio_autoparco from ap_viaggi where EXTRACT(YEAR FROM DATA_INSERIMENTO)=(SELECT EXTRACT(YEAR FROM SYSDATE) FROM DUAL) and autoparco=v('P1_AUTOPARCO') order by num_foglio_autoparco;
begin
open NumeroAutoparco;
loop
fetch NumeroAutoparco into FoglioAutoparco;
exit when NumeroAutoparco%notfound;
 if (FoglioAutoparco!=i+1)
 then 
     return i+1;
 else 
     i:=i+1;
 end if;
end loop;
close NumeroAutoparco;
select NVL(max(num_foglio_autoparco),0) into j from ap_viaggi where EXTRACT(YEAR FROM DATA_INSERIMENTO)=(SELECT EXTRACT(YEAR FROM SYSDATE) FROM DUAL) and autoparco=v('P1_AUTOPARCO');
return j+1;

end;
