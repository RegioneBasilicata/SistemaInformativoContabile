function CodiceGenerale return number as 
codice number;
FoglioGenerale number;
i number:=0;
j number:=0;
cursor NumeroGenerale is
select distinct num_foglio_viaggio from ap_viaggi where EXTRACT(YEAR FROM DATA_INSERIMENTO)=(SELECT EXTRACT(YEAR FROM SYSDATE) FROM DUAL) order by num_foglio_viaggio;
begin
open NumeroGenerale;
loop
fetch NumeroGenerale into FoglioGenerale;
exit when NumeroGenerale%notfound;
 if (FoglioGenerale!=i+1)
 then return i+1;
 else i:=i+1;
 end if;
end loop;
close NumeroGenerale;
select NVL(max(num_foglio_viaggio),0) into j from ap_viaggi where EXTRACT(YEAR FROM DATA_INSERIMENTO)=(SELECT EXTRACT(YEAR FROM SYSDATE) FROM DUAL) ;
return j+1;

end;
