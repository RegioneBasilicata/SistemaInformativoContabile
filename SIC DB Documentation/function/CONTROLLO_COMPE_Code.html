FUNCTION CONTROLLO_COMPE(p_anno in number,p_capitolo in VARCHAR2) RETURN NUMBER AS 
disp_capitolo number := 0;
pre_imp number := 0;
imp number := 0;

begin 
 select nvl(sum(nvl(open_balance_proposed,0)),0) 
 into pre_imp 
 from fin_t_documenti
 where segment8 = p_anno
 and segment2 = p_capitolo
 and doc_type = 'PRE-IMP'
 and nvl(deletion_status_flag,'N') = 'N'
 and nvl(open_balance_proposed,0)> 0;
 
 select nvl(sum(nvl(document_amount,0)),0) 
 into imp 
 from fin_t_documenti
 where segment8 = p_anno
 and segment2 = p_capitolo
 and doc_type like 'IMP%'
 and nvl(deletion_status_flag,'N') = 'N';
 
 disp_capitolo := previsione_definitiva(p_anno,p_capitolo) - pre_imp - imp;
 
 return(disp_capitolo);
END;
