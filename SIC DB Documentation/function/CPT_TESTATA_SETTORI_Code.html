function cpt_testata_settori (p3_anno in NUMBER, p3_eu in varchar2) return VARCHAR2 IS
testata varchar2(4000);
begin
 select 
 listagg(''''||descrizione,''',')
 within group (order by id) settori 
 into testata
 from cpt_settori 
 where id between (p3_anno * 100000) and ((p3_anno * 100000)+99999)
 and eu=p3_eu
 group by substr(id,1,4);

return (testata||'''');
end;
