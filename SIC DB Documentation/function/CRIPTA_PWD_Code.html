function      cripta_pwd (p_password in varchar2)
       return varchar2
       is
         l_password varchar2(4000);
       begin
       l_password := utl_raw.cast_to_raw(dbms_obfuscation_toolkit.md5
         (input_string => p_password));
       return l_password;
       end;
