function ctr_login_esterna (p_username in VARCHAR2, p_password in VARCHAR2)
  return BOOLEAN is
esiste number;
begin
SELECT count(*)
 into esiste  FROM fin_t_anagrafica
    Where (p_username=nvl(CODICE_FISCALE,'A') or
           p_username=nvl(PARTITA_IVA,'A')) and
           nvl(pass_temp,'-')=p_password;
update fin_t_anagrafica
  set pass_temp=null
  where nvl(pass_temp,'-')=p_password;
if Esiste=0
   then return false;
   else return true;
end if;
end;
