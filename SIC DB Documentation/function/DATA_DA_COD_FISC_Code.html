function data_da_cod_fisc (p_codice_fiscale in varchar2) return date
is

data_ns date;	

giorno varchar2(10);
mese varchar2(10);
anno varchar2(4);



begin


	select SUBSTR(p_codice_fiscale,7,2),
	CASE
	WHEN SUBSTR(p_codice_fiscale,9,1) = 'A' THEN '01'
	WHEN SUBSTR(p_codice_fiscale,9,1) = 'E' THEN '05'
	WHEN SUBSTR(p_codice_fiscale,9,1) = 'P' THEN '09'
	WHEN SUBSTR(p_codice_fiscale,9,1) = 'B' THEN '02'
	WHEN SUBSTR(p_codice_fiscale,9,1) = 'H' THEN '06'
	WHEN SUBSTR(p_codice_fiscale,9,1) = 'R' THEN '10'
	WHEN SUBSTR(p_codice_fiscale,9,1) = 'C' THEN '03'
	WHEN SUBSTR(p_codice_fiscale,9,1) = 'L' THEN '07'
	WHEN SUBSTR(p_codice_fiscale,9,1) = 'S' THEN '11'
	WHEN SUBSTR(p_codice_fiscale,9,1) = 'D' THEN '04'
	WHEN SUBSTR(p_codice_fiscale,9,1) = 'M' THEN '08'
	WHEN SUBSTR(p_codice_fiscale,9,1) = 'T' THEN '12'
  ELSE '13'
	END ,
	SUBSTR(p_codice_fiscale,10,2)
	into anno,mese,giorno
	from dual;

 if to_number(anno) > to_number(to_char(sysdate,'yy'))
  then anno := '19'||anno;
  else anno := '20'||anno;
 end if;  

 if to_number(giorno) > 40
  then giorno := to_number(giorno) - 40; 
 end if;
 
 if mese <> '13'
  then data_ns := to_date(giorno||mese||anno,'ddmmyyyy');
  return(data_ns);
  else return(null);
 end if;

END;
