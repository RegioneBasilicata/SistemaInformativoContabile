FUNCTION dati_catastali (p_codice number) RETURN varchar is
dati  varchar2(4000);
begin
for i in (SELECT foglio||'-'||particella||'-'||sub||'   ' dato
             from BI_CAT_FAB_PART P
             where cod_cesp=p_codice) loop
dati:=dati||i.dato;
end loop;
return dati;
end;
