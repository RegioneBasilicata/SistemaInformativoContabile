FUNCTION DISPONIBILITA_COMPE (p_anno in number,p_capitolo in VARCHAR2) RETURN NUMBER AS 
disp_capitolo number := 0;
visibilita_prev varchar2(1);
visibilita_assestamento varchar2(1);
BEGIN

  BEGIN
  select 
  ( nvl(PREVISTO_ATTUALE,0) 
  + nvl(VARIAZIONE_COMPETENZA_PIU,0)
  - nvl(VARIAZIONE_COMPETENZA_MENO,0)
  + nvl(ASSESTAMENTO_COMPETENZA_PIU,0)
  - nvl(ASSESTAMENTO_COMPETENZA_MENO,0)
  - nvl(PRE_IMPEGNI,0)
  - nvl(IMPEGNI_ACCERTAMENTI_COMPE,0)
  - DECODE(sign(nvl(COMPETENZA_BLOCCATA,0)),-1,0,nvl(COMPETENZA_BLOCCATA,0))
  - previsione_da_nascondere(p_anno, p_capitolo, 'N')
  - assestamento_da_nascondere(p_anno, p_capitolo, 'N')
  - blocco_automatico(p_anno, p_capitolo,'N')) into disp_capitolo
  from fin_t_bilancio
  where anno = p_anno
  and eu||ARTICOLO = p_capitolo;
  exception when OTHERS then disp_capitolo:=0;
  end;
 
  RETURN disp_capitolo;
END DISPONIBILITA_COMPE;
