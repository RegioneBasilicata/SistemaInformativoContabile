FUNCTION DISPONIBILITA_PERENTE 
(
  P_DOC_ID IN NUMBER 
 ,P_CAPITOLO IN VARCHAR2 
 ,P_ANNO IN NUMBER
) RETURN NUMBER AS 

disp number := 0;
open_balance number := 0;
disp_capitolo number := 0;

BEGIN
  begin
  SELECT NVL(d.open_balance_proposed,0) into open_balance
  FROM
  fin_t_documenti d
  WHERE
  d.doc_id = p_doc_id;
  exception when no_data_found then open_balance := 0;
  end;
  
  disp_capitolo := nvl(disponibilita_compe(p_anno,p_capitolo),0);
  
  if nvl(open_balance,0) >= nvl(disp_capitolo,0)
   then disp := nvl(disp_capitolo,0);
   else disp := nvl(open_balance,0);
  end if; 
  RETURN disp;
END DISPONIBILITA_PERENTE;
