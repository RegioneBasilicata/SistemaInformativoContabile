function documenti_pluriennali(p_anno in NUMBER, p_capitolo in varchar2,p_tipo varchar2) return number IS
/*
Lino 19.01.2018

restituisce la somma per tipo documento (ammesso nel pluriennale) di quelli emessi come pluriennale;
I parametri sono:
Anno     : L'anno per il quale avere il valore
Capitolo : Capitolo per il quale si desidera conoscere il pluriennale
Tipo     : Può valere PRE-IMP o IMP-DEF

Se il documento è un pre-imp, devo riportare il reale residuo calcolandolo sottraendogli il valore degli
impegni definitivi se quest'ultimi sono stati generati al primo gennaio dell'anno in considerazione.

*/

importo number(15,2) := 0; 
appoggio_residuo number(15,2);
appoggio_importo number(15,2);
appoggio_doc_id number;

cursor preimpegni is
 select 
 doc_id,
 document_amount
 from fin_t_documenti d
 where d.segment8 = p_anno
 and d.segment2 = p_capitolo
 and d.doc_type = 'PRE-IMP'
 and to_char(date_created,'ddmmyyyy')='0101'||to_char(p_anno)
 and nvl(deletion_status_flag,'N')<>'Y';

BEGIN
if p_tipo <>'PRE-IMP'
   then
   select sum(nvl(document_amount,0))
   into appoggio_importo 
   from fin_t_documenti d
   where d.segment8 = p_anno
   and d.segment2 = p_capitolo
   and d.doc_type = p_tipo
   and to_char(date_created,'ddmmyyyy')='0101'||to_char(p_anno)
   and nvl(deletion_status_flag,'N')<>'Y';
   else 
   appoggio_importo:=0;
   for a in preimpegni loop
    select sum(document_amount)
    into appoggio_residuo
    from fin_t_documenti
    where doc_type='IMP-DEF'
    and previous_doc_id = a.doc_id
    and to_char(date_created,'ddmmyyyy')='0101'||to_char(p_anno)
    and nvl(deletion_status_flag,'N')<>'Y';
    appoggio_importo := appoggio_importo + (a.document_amount - nvl(appoggio_residuo,0));
   end loop;
end if;   
 return nvl(appoggio_importo,0);
END;
