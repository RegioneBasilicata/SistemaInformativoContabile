function economie_anno (p_anno number,p_doc_id number) return number as

totale_eco number := 0;

begin

select nvl(sum(nvl(document_amount,0)),0) into totale_eco
from fin_t_documenti
where doc_type in ('ECO','DIS')
and segment8 = p_anno
and previous_doc_id = p_doc_id;

return(totale_eco);

end;
