function ELENCO_BENEFICIARI (
   p_anno in number,
   p_doc_id in number
) return varchar2
is

cursor beneficiari is
select 
decode(tipo_anagrafica,'F',fta.cognome||' '||fta.nome,fta.denominazione) beneficiario
,ftrub.importo importo
from FIN_T_RIACCERTAMENTI_UFF_BEN ftrub
   ,FIN_T_ANAGRAFICA fta
where 
ftrub.ANNO_BILANCIO_RIACCERTAMENTO = p_anno
   and ftrub.DOC_ID = p_doc_id
   and fta.ID = ftrub.ID_ANA;

welenco varchar2(4000);

begin



   for i in beneficiari loop
      welenco := welenco || i.beneficiario || ' - '|| to_char(i.importo,'999G999G999G999G990D00') || ' / ';
      if length(nvl(welenco,'0')) > 3500 
         then return(welenco);
      end if;
   end loop;

   if welenco is not null then
      welenco := substr(welenco, 1, length(welenco)-2);
   end if;
   return welenco;

end;
