FUNCTION forn_contr (p_tab_codice number, p_progr NUMBER)
RETURN varchar is
dati  varchar2(4000);
begin
for i in (SELECT SUBSTR(DECODE(fa.TIPO_ANAGRAFICA,'F',fa.COGNOME||' '||fa.nome,fa.DENOMINAZIONE),1,2000) ragione_sociale
             from bi_fornitori_contratti bf,fin_t_anagrafica fa
             where bf.tab_codice = p_tab_codice and
                   bf.gen_progr= p_progr and
                   fa.id = bf.fornitore) loop
if length(dati||rtrim(i.ragione_sociale)||' - ')<4000 then
  dati:=dati||rtrim(i.ragione_sociale)||' - ';
end if; 
end loop;
return dati;
end;
