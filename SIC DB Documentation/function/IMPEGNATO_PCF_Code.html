FUNCTION IMPEGNATO_PCF (p_anno in number,p_pcf in VARCHAR2) RETURN NUMBER AS 
importo_impegnato number := 0;

begin
select nvl(sum(nvl(d.document_amount,0)),0) into importo_impegnato
from fin_t_documenti d
where d.doc_type like 'IMP%'
and nvl(d.deletion_status_flag,'N') = 'N'
and d.segment8 = p_anno
and d.attribute16 = p_pcf;

RETURN importo_impegnato;
END;
