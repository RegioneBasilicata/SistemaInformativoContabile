FUNCTION IMPEGNATO_TP (p_anno in number,p_id_ana in number) RETURN NUMBER AS 
importo_impegnato number := 0;

begin
select nvl(sum(nvl(t.amount,0)),0) into importo_impegnato
from fin_t_documenti d
,fin_t_documenti_terze_parti t
where d.doc_type in ('IMP','IMP-PER','IMP-DEF')
and nvl(d.deletion_status_flag,'N') = 'N'
and d.segment8 = p_anno
and t.doc_id = d.doc_id
and t.id_ana = p_id_ana;

RETURN importo_impegnato;
END;
