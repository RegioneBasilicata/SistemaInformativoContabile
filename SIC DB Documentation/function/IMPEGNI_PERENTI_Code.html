function impegni_perenti (p_anno in NUMBER, p_capitolo in varchar2) return number IS
tot_impegni_per number;
begin
select nvl(sum(document_amount),0) into tot_impegni_per
from fin_t_documenti
where doc_type = 'IMP-PER'
and segment8 = p_anno
--and ATTRIBUTE21 is null 
--and DATE_CREATED > to_date('0101'||segment8,'ddmmyyyy')
and nvl(deletion_status_flag,'N') = 'N'
and segment2 = p_capitolo;

return(tot_impegni_per);
end;
