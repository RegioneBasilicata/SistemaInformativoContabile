function importo_blocco (p_anno in number, p_capitolo in varchar2, p_cassa varchar2) return number is

	imp_blocco number := 0;

	begin
	 if nvl(p_cassa,'N') = 'N'
	 then
	 select nvl(sum(decode(doc_type,'BLOCCO_COMPE',nvl(document_amount,0),-nvl(document_amount,0))),0) into imp_blocco
	 from fin_t_documenti
	 where doc_type like '%BLOCCO_COMPE'
	 and segment8 = p_anno
	 and segment2 = p_capitolo
	 and nvl(deletion_status_flag,'N') = 'N';
	 else
	 select nvl(sum(decode(doc_type,'BLOCCO_CASSA',nvl(document_amount,0),-nvl(document_amount,0))),0) into imp_blocco
	 from fin_t_documenti
	 where doc_type like '%BLOCCO_CASSA'
	 and segment8 = p_anno
	 and segment2 = p_capitolo
	 and nvl(deletion_status_flag,'N') = 'N';
	 end if;

	 return(imp_blocco);

	end;
