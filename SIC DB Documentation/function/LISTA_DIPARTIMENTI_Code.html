function lista_dipartimenti(a_bene_imm in VARCHAR2, 
                                     a_stanza in NUMBER) return varchar2 IS
dip varchar2(4000);                                     
begin                                     
   select listagg(dip,chr(13)||chr(10)) within group (ORDER BY dip) dipartimenti 
   into dip from
   (select distinct st.DIPARTIMENTO||' - '||st.DESC_DIPARTIMENTO dip
  from tb_stanze_occ o,vw_dipartimenti_strutture st
   where o.data_fine_occ is null and
         o.stanza_bene_imm=a_bene_imm and
         o.stanza_progr=a_stanza and
         st.DIPARTIMENTO=substr(ass_ufficio,1,2)) ;
return dip;
end;
