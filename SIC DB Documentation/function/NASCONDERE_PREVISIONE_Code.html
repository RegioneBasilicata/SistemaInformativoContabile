function       nascondere_previsione(p_anno in NUMBER, p_capitolo in varchar2, p_cassa varchar2) return number IS
importo number(15,2) := 0; 
anno0 number(4);
anno1 number(4);
anno2 number(4);
provvisorio_aperto varchar2(1);
visibilita_previsione varchar2(1);
visibilita_provvisorio varchar2(1);
visibilita varchar2(1);
anno_appoggio number(4);

BEGIN


 begin
  select 
    anno_previsione1
   ,anno_previsione2
   ,anno_previsione3
   ,bilancio_previsione_visibile
   ,bilancio_provvisorio_visibile
   ,bilancio_provvisorio_aperto
   ,to_number(to_char(sysdate,'yyyy'))
  into 
   anno0
  ,anno1
  ,anno2
  ,visibilita_previsione
  ,visibilita_provvisorio
  ,provvisorio_aperto
  ,anno_appoggio
  from fin_t_dati_generali2;
  exception when no_data_found then 
    visibilita_previsione  :='N';
    visibilita_provvisorio :='N';
    provvisorio_aperto     :='N';
    anno0 := anno_appoggio;
    anno1 :=anno_appoggio+1;
    anno2 :=anno_appoggio+2;
 end;

/* Lino 9.1.2018
Solo per la cassa è necessario verificare che se il provvisorio è visibile ed attivo, allora la visibilità deve essere ammessa al fine di permettere i pagamenti
infatti la cassa, anche come provvisorio viene generata come documento B_CAS.
*/
if p_cassa='S' and 
   visibilita_provvisorio='S' and
   provvisorio_aperto='S' 
   then 
   visibilita:='S';
   else visibilita := visibilita_previsione; 
end if ;  

if visibilita = 'N' and
   p_anno = anno0
   and p_cassa = 'N'
   then 
   select 
   nvl(sum(decode(substr(d.doc_type,7,3),'NEG',(d.document_amount * -1),d.document_amount)),0) 
   into importo
   from  fin_t_documenti d
   where d.segment8 = p_anno
   and d.segment2 = p_capitolo
   and  d.doc_type in 
   ('B_COM_POS'
   ,'B_COM_NEG'
   );
end if;

if visibilita = 'N' and
   p_anno = anno0
   and p_cassa = 'S'
   then 
   select 
   nvl(sum(decode(substr(d.doc_type,7,3),'NEG',(d.document_amount * -1),d.document_amount)),0) 
   into importo
   from  fin_t_documenti d
   where d.segment8 = p_anno
   and d.segment2 = p_capitolo
   and  d.doc_type in 
   ('B_CAS_POS'
   ,'B_CAS_NEG'
   );
end if;

if visibilita = 'N' and
   p_anno = anno1
   then 
   select 
   nvl(sum(decode(substr(d.doc_type,7,3),'NEG',(d.document_amount * -1),d.document_amount)),0) 
   into importo
   from  fin_t_documenti d
   where d.segment8 = p_anno
   and d.segment2 = p_capitolo
   and  d.doc_type in 
   ('B_PL1_POS'
   ,'B_PL1_NEG'
   );
end if;

if visibilita = 'N' and
   p_anno = anno2
   then 
   select 
   nvl(sum(decode(substr(d.doc_type,7,3),'NEG',(d.document_amount * -1),d.document_amount)),0) 
   into importo
   from  fin_t_documenti d
   where d.segment8 = p_anno
   and d.segment2 = p_capitolo
   and  d.doc_type in 
   ('B_PL2_POS'
   ,'B_PL2_NEG'
   );
end if;

 return nvl(importo,0);
END;
