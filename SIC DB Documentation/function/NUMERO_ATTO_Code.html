FUNCTION NUMERO_ATTO (p_doc_id number) RETURN varchar2 IS

preimp_sn varchar2(1);
tipo_doc varchar2(10);
tipo_atto varchar2(100);
numero_atto varchar2(100);
numero_atto_pre varchar2(100);
p_attribute23 varchar2(100);


begin 
 select doc_type 
 ,nvl(attribute13,'X')
 ,nvl(attribute23,'X')
 ,attribute14
 into tipo_doc
 ,tipo_atto
 ,p_attribute23
 ,numero_atto
 from fin_t_documenti
 where doc_id = p_doc_id;

 if (tipo_doc = 'PRE-IMP' and tipo_atto = 'DETERMINA' and p_attribute23 = 'X')
  then 
     begin
     select i.attribute14 into numero_atto_pre
     from fin_t_documenti i 
     where i.doc_type like 'IMP%' 
     and i.previous_doc_id = p_doc_id 
     and i.doc_id = (select max(doc_id) from fin_t_documenti where doc_type like 'IMP%' and previous_doc_id = p_doc_id);
     exception when no_data_found then numero_atto_pre := numero_atto;
     end;
     return(numero_atto_pre);
  else
    return(numero_atto);   
 end if;



End;
