FUNCTION OPEN_BALANCE_ACC_TP (p_doc_id number, p_id_ana number, p_id_ana_sede number) RETURN number IS

tot_accertamenti number := 0;
totale_tp number := 0;

begin

   select nvl(sum(nvl(t.amount,0)),0) into tot_accertamenti
   from fin_t_documenti d
   ,fin_t_documenti_terze_parti t
   where d.doc_type = 'REV'
   and d.previous_doc_id = p_doc_id
   and nvl(d.deletion_status_flag,'N') = 'N'
   and nvl(d.open_balance_proposed,0) > 0
  -- and d.data_created <= p_data
   and t.doc_id = d.doc_id
   and t.id_ana = p_id_ana
   and t.id_sede = p_id_ana_sede;

  begin
   select nvl(sum(nvl(amount,0)),0) into totale_tp
   from fin_t_documenti_terze_parti
   where doc_id = p_doc_id
   and id_ana = p_id_ana
   and id_sede = p_id_ana_sede;
 -- exception when no_data_found then totale_tp := 0;
  end;

  return(totale_tp - tot_accertamenti);

end;
