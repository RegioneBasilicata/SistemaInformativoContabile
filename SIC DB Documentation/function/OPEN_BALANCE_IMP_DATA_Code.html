FUNCTION OPEN_BALANCE_IMP_DATA (p_doc_id number, p_data date) RETURN number IS

tot_liq number := 0;
totale_imp number := 0;

begin

   select nvl(sum(nvl(d.DOCUMENT_AMOUNT,0)),0) into tot_liq
   from fin_t_documenti d
   where d.doc_type in ('LIQ','ECO','DIS','IMP-PER')
   and d.previous_doc_id = p_doc_id
   and nvl(d.DELETION_STATUS_FLAG,'N') = 'N'
   and d.date_created <= p_data;


   select nvl(sum(nvl(document_amount,0)),0) into totale_imp
   from fin_t_documenti
   where doc_id = p_doc_id;

  return(totale_imp - tot_liq);

end;
