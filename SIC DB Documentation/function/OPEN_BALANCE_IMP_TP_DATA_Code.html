FUNCTION OPEN_BALANCE_IMP_TP_DATA (p_doc_id number, p_id_ana number, p_id_ana_sede number,p_data date) RETURN number IS

tot_liquidazioni number := 0;
tot_eco number := 0;
totale_tp number := 0;
doc_amount number := 0;
residuo_doc number := 0;

begin

 select open_balance_imp_data(p_doc_id,p_data) into residuo_doc
 from dual;
 
 if residuo_doc > 0 then

   select nvl(sum(nvl(d.document_amount,0)),0) into doc_amount
   from fin_t_documenti d
   where d.doc_id = p_doc_id;

   select nvl(sum(nvl(t.amount,0)),0) into tot_liquidazioni
   from fin_t_documenti d
   ,fin_t_documenti_terze_parti t
   where d.doc_type in ('IMP-PER','LIQ','ECO','DIS')
   and d.previous_doc_id = p_doc_id
   and nvl(d.deletion_status_flag,'N') = 'N'
   --and nvl(d.open_balance_proposed,0) > 0
   and d.date_created <= p_data
   and t.doc_id = d.doc_id
   and t.id_ana = p_id_ana
   and t.id_sede = p_id_ana_sede;

   select nvl(sum(nvl(amount,0)),0) into totale_tp
   from fin_t_documenti_terze_parti
   where doc_id = p_doc_id
   and id_ana = p_id_ana
   and id_sede = p_id_ana_sede;
 
   if doc_amount = totale_tp
    then 
       select nvl(sum(nvl(d.document_amount,0)),0) into tot_eco
       from fin_t_documenti d
       where d.doc_type in ('ECO','DIS')
       and nvl(d.deletion_status_flag,'N') = 'N'
       and d.previous_doc_id = p_doc_id
       and d.date_created <= p_data
       and d.doc_id not in (select t.doc_id from fin_t_documenti d
                               ,fin_t_documenti_terze_parti t
                               where d.doc_type in ('ECO','DIS')
                               and d.previous_doc_id = p_doc_id
                               and nvl(d.deletion_status_flag,'N') = 'N'
                               and t.doc_id = d.doc_id
                               and t.id_ana = p_id_ana
                               and t.id_sede = p_id_ana_sede);
        return(totale_tp - tot_liquidazioni - tot_eco);
   else return(totale_tp - tot_liquidazioni);
   end if;
   else return(0);
   end if;
   
end;
