FUNCTION OPEN_BALANCE_LIQ_DATA (p_doc_id number, p_data date) RETURN number IS

tot_mandati number := 0;
totale_liq number := 0;

begin

   select nvl(sum(nvl(d.open_balance_proposed,0)),0) into tot_mandati
   from fin_t_documenti d
   where d.doc_type = 'MAND'
   and d.previous_doc_id = p_doc_id
   and nvl(d.open_balance_proposed,0) > 0
   and d.date_created <= p_data;


   select nvl(sum(nvl(document_amount,0)),0) into totale_liq
   from fin_t_documenti
   where doc_id = p_doc_id;

  return(totale_liq - tot_mandati);

end;
