function ORDINA_VOCI_MENU_ASSEGNATE (
   p_codice number,
   p_orfani varchar2
) return number 
is

w_return number;
w_app number;

begin

    w_app := 0;
    
for i in (
select SIC_ALBERO_MENU(codice)||' '||TITOLO as display_value, CODICE as return_value
  from FIN_T_MENU_WEB
  where (p_orfani = 'N'
  and codice in (select codice from FIN_T_MENU_RUOLI_WEB)) 
  or (p_orfani = 'S'
  and codice not in (select codice from FIN_T_MENU_RUOLI_WEB))
start with codice_padre is null
connect by prior codice = codice_padre
order siblings by ordinamento,titolo
) loop

   w_app := w_app + 1; 
    
   if i.return_value = p_codice then
      w_return := w_app ;
   end if;

end loop;

return w_return;
end;
