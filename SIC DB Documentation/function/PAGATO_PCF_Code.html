FUNCTION PAGATO_PCF (p_anno in number,p_pcf in varchar2) RETURN NUMBER AS 
importo_pagato number := 0;

begin
select nvl(sum(nvl(d.open_balance_proposed,0)),0) into importo_pagato
from fin_t_documenti d
where d.doc_type = 'MAND'
and nvl(d.open_balance_proposed,0) > 0
and d.segment8 = p_anno
and d.attribute16 = p_pcf;

RETURN importo_pagato;
END;
