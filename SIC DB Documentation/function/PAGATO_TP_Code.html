FUNCTION PAGATO_TP (p_anno in number,p_id_ana in number) RETURN NUMBER AS 
importo_pagato number := 0;

begin
select nvl(sum(nvl(t.amount,0)),0) into importo_pagato
from fin_t_documenti d
,fin_t_documenti_terze_parti t
where d.doc_type = 'MAND'
and nvl(d.open_balance_proposed,0) > 0
and d.segment8 = p_anno
and t.doc_id = d.doc_id
and t.id_ana = p_id_ana;

RETURN importo_pagato;
END;
