FUNCTION PREIMPEGNATO_UFFICIO (p_anno in number,p_capitolo in VARCHAR2, p_ufficio in VARCHAR2) RETURN NUMBER AS 
importo_preimpegnato number := 0;

begin
select nvl(sum(nvl(d.open_balance_proposed,0)),0) into importo_preimpegnato
from fin_t_documenti d
where d.doc_type = 'PRE-IMP'
and nvl(d.deletion_status_flag,'N') = 'N'
and d.segment8 = p_anno
and d.segment3 = p_ufficio
and d.segment2 = p_capitolo;

RETURN importo_preimpegnato;
END;
