function preimpegni_aperti(p_anno in NUMBER, p_capitolo in varchar2,p_tipo_importo in varchar2,p_tipo_atto in varchar2) return number IS

importo number(15,2) :=0;

BEGIN

 select 
  decode(p_tipo_importo,'D',sum(document_amount),sum(open_balance_proposed))
  into importo
  from fin_t_documenti
  where segment8=p_anno
  and segment2=p_capitolo
  and nvl(attribute13,'PR') = p_tipo_atto
  and doc_type = 'PRE-IMP' 
  and nvl(deletion_status_flag,'N') <>'Y' 
  and open_balance_proposed <>0 
  and attribute23  is not null -- identifica i preimpegni effettivi, non quelli generati da provedimenti in modo provvisorio
  group by segment2,attribute13;
 return nvl(importo,0);
END;
