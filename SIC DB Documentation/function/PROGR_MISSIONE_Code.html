function progr_missione (p_matricola in NUMBER, p_periodo in date)
         return NUMBER
         Is
p_progr number;         
Begin
Select Nvl(Max(Progr),0)+1
       into p_progr
       From Ap_Missioni 
       Where Matricola=P_Matricola And
             Periodo=P_Periodo;
Return (p_progr);
End;
