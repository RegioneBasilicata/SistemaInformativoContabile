function progr_passeggero (viaggio in number,num_pass in number)
return number as 
a_ind number;
a_progr number; 
begin
a_progr:=0;
a_ind:=1;
for p in (select progr
           from ap_pass_viaggio
           where codice_viaggio=viaggio
           order by progr) loop
if a_ind=num_pass then
--   a_progr:=p.progr;
   return p.progr;
else a_ind:=a_ind+1;
end if;
end loop;
return 0;
end;
