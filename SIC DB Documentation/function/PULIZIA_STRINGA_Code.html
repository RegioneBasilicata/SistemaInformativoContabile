function pulizia_stringa(vvv varchar2) return varchar2 is
   a_vvv varchar2(4000);
  begin
 -- potrebbe essere usata anche select CONVERT('vvv, 'US7ASCII', 'UTF8') FROM DUAL; 
    a_vvv:=vvv;
    a_vvv := replace(a_vvv,'°','');
    a_vvv := replace(a_vvv,'^','');
    a_vvv := replace(a_vvv,'à','a');
    a_vvv := replace(a_vvv,'è','e');
    a_vvv := replace(a_vvv,'é','e');
    a_vvv := replace(a_vvv,'ù','u');
    a_vvv := replace(a_vvv,'ò','');
    a_vvv := replace(a_vvv,';',',');
    a_vvv := replace(a_vvv,'"','');
   -- a_vvv := replace(a_vvv,chr(13),'');   
    a_vvv := REPLACE(REPLACE(a_vvv, CHR(13), ' '), CHR(10), ' ');
    
    return (a_vvv);
  end;
