FUNCTION readfromweb (url VARCHAR2)
   RETURN CLOB
IS
   pcs    UTL_HTTP.html_pieces;
   retv   CLOB;
BEGIN
   
   pcs := UTL_HTTP.request_pieces (url, 32767);
   FOR i IN 1 .. pcs.COUNT
   LOOP
      retv := retv || pcs (i);
   END LOOP;
   RETURN retv;
END;
