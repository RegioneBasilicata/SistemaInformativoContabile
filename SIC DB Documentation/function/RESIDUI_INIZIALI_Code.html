TEXT 
FUNCTION      residui_iniziali (p_anno number,p_capitolo varchar2) RETURN NUMBER IS
return_value number(15,2) := 0;
-- procedura U_CONSUNTIVO_PERENTI (UCP) 
UCP_w_residuo_consuntivo number := 0;
UCP_w_man_rev_residui number := 0;
UCP_w_variazioni_residui number := 0;
UCP_w_per_anno1 number := 0;
UCP_w_per_anno2 number := 0;
UCP_w_per_anno3 number := 0;
-- procedura U_CONSUNTIVO_PERENTI_SUCC (UCPS) 
UCPS_w_man_rev_residui number := 0;
UCPS_w_pag_per_anno1 number := 0;
UCPS_w_pag_per_anno2 number := 0;
UCPS_w_pag_per_anno3 number := 0;
-- procedura U_CONSUNTIVO_RESIDUI (UCR)
UCR_w_residuo_consuntivo number := 0;
UCR_w_man_rev_residui number := 0;
UCR_w_variazioni_residui number := 0;
UCR_w_pag_anno1 number := 0;
UCR_w_pag_anno2 number := 0;
UCR_w_pag_anno3 number := 0;
UCR_w_residuo_perenzione number := 0;
-- FINE varibili utilizzate per inserire il risultato delle stored procedure
data_generazione date:= to_date('3112'||p_anno,'ddmmyyyy');



BEGIN

gat2.U_CONSUNTIVO_RESIDUI(p_anno,p_capitolo,UCR_w_residuo_consuntivo,UCR_w_man_rev_residui,UCR_w_variazioni_residui,UCR_w_pag_anno1,UCR_w_pag_anno2,UCR_w_pag_anno3,UCR_w_residuo_perenzione,data_generazione);
gat2.U_CONSUNTIVO_PERENTI_SUCC(p_anno,p_capitolo,UCPS_w_man_rev_residui,UCPS_w_pag_per_anno1,UCPS_w_pag_per_anno2,UCPS_w_pag_per_anno3,data_generazione);
gat2.U_CONSUNTIVO_PERENTI(p_anno,p_capitolo,UCP_w_residuo_consuntivo,UCP_w_man_rev_residui,UCP_w_variazioni_residui,UCP_w_per_anno1,UCP_w_per_anno2,UCP_w_per_anno3,data_generazione);

return_value := nvl(UCR_w_residuo_consuntivo,0) + nvl(UCPS_w_man_rev_residui,0) + nvl(UCP_w_residuo_consuntivo,0);

return return_value;
END;
