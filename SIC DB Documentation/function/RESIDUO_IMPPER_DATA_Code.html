FUNCTION RESIDUO_IMPPER_DATA (p_doc_id number, p_data date) RETURN number IS

  importo_doc NUMBER := 0;
  TOT_STORNO NUMBER := 0;
  TOT_MAND NUMBER := 0;
  TOT_IMPPER NUMBER := 0;
  TOT_ECO_PERENTE number := 0;
  TOT_ECO_IMPPER number := 0;


BEGIN
 select nvl(document_amount,0) into importo_doc
 from fin_t_documenti 
 where doc_id = p_doc_id;

  	SELECT
		nvl(sum(nvl(jugd2.document_amount,0)),0) 	into TOT_IMPPER
		FROM
		fin_t_documenti 					jugd2
		WHERE
		jugd2.doc_type 	= 'IMP-PER' 	and
		nvl(jugd2.deletion_status_flag,'N') <> 'Y' 	and
		jugd2.date_created 		<= p_data	and
		jugd2.previous_doc_id	= p_doc_id;

    SELECT SUM(nvl(jugd2.open_balance_proposed,0)) INTO TOT_ECO_PERENTE
		FROM 
			fina_documenti 		jugd1,
			fina_documenti 		jugd2
		WHERE
			jugd2.doc_type in ('ECO','DIS')  		and
			jugd1.doc_type = 'IMP-PER' 		and
			jugd2.previous_doc_id = jugd1.doc_id  and
			nvl(jugd2.proposal_status_flag,'Z')||nvl(jugd2.auditing_status_flag,'Z')	='PI' and
			nvl(jugd2.deletion_status_flag,'N') <> 'Y' 		and
      jugd2.date_created <= p_data and 
			jugd1.previous_doc_id = p_doc_id AND
      jugd1.date_created <= p_data;

    SELECT SUM(nvl(jugd2.open_balance_proposed,0)) INTO TOT_ECO_IMPPER
		FROM 
			fina_documenti 		jugd1,
			fina_documenti 		jugd2
		WHERE
			jugd2.doc_type = 'ECO'  		and
			jugd1.doc_type = 'IMP-PER' 		and
			jugd2.previous_doc_id = jugd1.doc_id  and
			nvl(jugd2.proposal_status_flag,'Z')||nvl(jugd2.auditing_status_flag,'Z')	='PI' and
			nvl(jugd2.deletion_status_flag,'N') <> 'Y' 		and
      (jugd2.date_created >= to_Date('01012005','ddmmyyyy') and jugd2.date_created <= p_data) and 
			jugd1.previous_doc_id = p_doc_id AND
      jugd1.date_created <= p_data;

    SELECT
		nvl(sum(nvl(jugd2.open_balance_proposed,0)),0) 	into tot_storno
		FROM
		fin_t_documenti 					jugd2
		WHERE
		jugd2.doc_type 	in ('ST-IMP','DIS','ECO','ST-IMPDEF') 	and
		nvl(jugd2.deletion_status_flag,'N') <> 'Y' 	and
		jugd2.date_created 		<= p_data	and
		jugd2.previous_doc_id	= p_doc_id;

    SELECT 
		nvl(sum(nvl(jugd.open_balance_proposed,0)),0) into 	tot_mand
		FROM
		fina_documenti 		jugd--,
		WHERE
		jugd.doc_type = 'MAND' and
    nvl(jugd.open_balance_proposed,0) > 0 and
		jugd.date_created 	<= p_data	and
		jugd.doc_id_capo_filiera = p_doc_id;




     return (importo_doc - NVL(TOT_IMPPER,0)- NVL(TOT_STORNO,0) - NVL(TOT_MAND,0) - NVL(TOT_ECO_IMPPER,0) + NVL(TOT_ECO_PERENTE,0));

END RESIDUO_IMPPER_DATA;
