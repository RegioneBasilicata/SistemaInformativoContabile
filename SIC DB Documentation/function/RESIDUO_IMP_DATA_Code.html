FUNCTION RESIDUO_IMP_DATA (p_doc_id number, p_data date) RETURN number IS

  importo_doc NUMBER := 0;
  TOT_STORNO NUMBER := 0;
	TOT_MAND NUMBER := 0;
    verifica_pluriennale number := 0; -- mi serve per capire se l'IMP è pluriennale o no


BEGIN
    select count(*) into verifica_pluriennale
    from fin_t_documenti 
    where doc_id = p_doc_id
    and date_created = to_date('0101'||segment8,'ddmmyyyy');
    
    if verifica_pluriennale > 0
        then
            begin
             select nvl(document_amount,0) into importo_doc
             from fin_t_documenti 
             where doc_id = p_doc_id
             and (nvl(to_date(attribute38,'yyyymmdd'),date_created) <= p_data or attribute21 is not null);
             exception when no_data_found then importo_doc := 0;
             end;
        else 
           begin
             select nvl(document_amount,0) into importo_doc
             from fin_t_documenti 
             where doc_id = p_doc_id
             and date_created <= p_data;
             exception when no_data_found then importo_doc := 0;
             end;
     end if;
     

 if importo_doc > 0
 then

    SELECT
		nvl(sum(nvl(jugd2.open_balance_proposed,0)),0) 	into tot_storno
		FROM
		fin_t_documenti 					jugd2
		WHERE
		jugd2.doc_type 	in ('ST-IMP','DIS','ECO','ST-IMPDEF') 	and
		nvl(jugd2.deletion_status_flag,'N') <> 'Y' 	and
		jugd2.date_created 		<= p_data	and
		jugd2.previous_doc_id	= p_doc_id;

    SELECT 
		nvl(sum(nvl(jugd.open_balance_proposed,0)),0) into 	tot_mand
		FROM
		fina_documenti 		jugd--,
		WHERE
		jugd.doc_type = 'MAND' and
        nvl(jugd.open_balance_proposed,0) > 0 and
		jugd.date_created 	<= p_data	and
		jugd.doc_id_capo_filiera = p_doc_id;


     return (importo_doc - NVL(TOT_STORNO,0) - NVL(TOT_MAND,0));
     
     else return(0);
     end if;

END RESIDUO_IMP_DATA;
