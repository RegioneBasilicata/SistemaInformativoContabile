FUNCTION REVERSALI_TP(p_doc_id in number) return number is
-- funzione per verificare se i beneficiari del mandato sono uguali a quelli delle rev collegate allora return = 0 in caso contrario return > 0
tp number := 0;
importo_rev number := 0;

cursor mand is
select d.doc_sequence_value num_mand
,t.id_ana m_id_ana
,t.id_sede m_id_sede
,t.amount importo_tp
from FIN_T_DOCUMENTI_TERZE_PARTI t
,FIN_T_DOCUMENTI d
where d.doc_id = p_doc_id
and d.doc_type = 'MAND'
and t.doc_id = d.doc_id;


begin 
 tp := 0;
  for m in mand loop
   select nvl(sum(nvl(t.amount,0)),0) into importo_rev
   from fin_t_documenti_terze_parti t
   ,fin_t_documenti d
   ,FIN_T_REV_COLL r
   where r.REV_MAND_COD = m.num_mand
   and d.doc_type = 'REV'
   and d.doc_sequence_value = r.rev_codice
   and t.doc_id = d.doc_id
   and t.id_ana = m.m_id_ana
   and t.id_sede = m.m_id_sede;
  
  if importo_rev <> nvl(m.importo_tp,0)
   then tp := tp + 1;
  end if;
 end loop; 
 return(tp);
end;
