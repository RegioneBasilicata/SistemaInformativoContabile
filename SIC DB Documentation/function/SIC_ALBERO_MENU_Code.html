function      SIC_ALBERO_MENU (wcodice number) return varchar2 is

wcodice_padre number;

begin

begin
select codice_padre
into wcodice_padre
from fin_t_menu_web
where codice = wcodice;
end;

if wcodice_padre is null then return '...';
else return '...'||SIC_ALBERO_MENU(wcodice_padre);
end if;

end;
