function situazione_km_cc (w_cc in number) return number is
 da_data date;
 a_data  date;
 tot_km   number;
begin
select da_periodo, a_periodo
   into da_data,a_data
   from ap_centri_costo
   where codice=w_cc;
-- calcolo spese viaggi generati da richieste 
select round(sum((nvl(V.KM_FINE,V.KM_INZIO)-V.KM_INZIO)*nvl(p.percentuale,100)/100),0)
  into tot_km
  from ap_viaggi v,vw_perc_cc_viaggio p
  where v.data_partenza between da_data and a_data and
        v.centro_costo=w_cc and
        nvl(v.escludi_calcolo_cc,'N')='N' and
        p.codice_viaggio(+)=v.codice and
        P.CENTRO_COSTO(+)=v.centro_costo;
return nvl(tot_km,0);
end;
