function      stato_documento(p_doc_id in NUMBER) return varchar2 IS

tipo_documento varchar2(10);
stato_documento varchar2(50);
storno number;

BEGIN

 begin
  select doc_type,
  decode(nvl(deletion_status_flag,'N'),'Y','Cancellato','Valido')
  into 
  tipo_documento,
  stato_documento
  from fin_t_documenti 
  where doc_id = p_doc_id;
  exception when no_data_found then stato_documento := 'Non Rilevato';
 end;

 if tipo_documento in ('MAND','REV')
    then 
    begin 
     select doc_sequence_value
     into storno
     from  fin_t_documenti d
     where previous_doc_id = p_doc_id
     and doc_type in ('ST-MAND','ST-REV');
     exception when no_data_found then storno := null;
    end;
 end if;

 if storno is not null
    then stato_documento :='Storno n.'||storno;
 end if;

 return stato_documento;
END;
