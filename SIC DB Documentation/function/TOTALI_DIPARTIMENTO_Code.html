function totali_dipartimento (w_tipo in VARCHAR2, 
                                         w_dipartimento in VARCHAR2, 
                                         w_anno in NUMBER,
                                         w_corr_cap in number
                                         ) return number IS
totale number := 0;

begin
if upper(w_tipo) = 'IMP'
then
  begin
  select nvl(sum(nvl(i.document_amount,0)),0) into totale
    from fin_t_documenti i 
    ,fin_t_articoli a 
    where i.doc_type like upper(w_tipo)||'%'
    and nvl(i.deletion_status_flag,'N') = 'N'
    and i.segment8 = w_anno
    and nvl(substr(i.segment3,1,2),'X') = w_dipartimento
    and a.anno = i.segment8
    and a.eu||a.codice = i.segment2
    and a.nb_titolo = w_corr_cap;
    exception when no_data_found then totale := 0;
  end;
 else  
  begin
  select nvl(sum(nvl(i.OPEN_BALANCE_PROPOSED,0)),0) into totale
    from fin_t_documenti i 
    ,fin_t_articoli a 
    where i.doc_type like upper(w_tipo)||'%'
    and nvl(i.deletion_status_flag,'N') = 'N'
    and i.segment8 = w_anno
    and nvl(substr(i.segment3,1,2),'X') = w_dipartimento
    and a.anno = i.segment8
    and a.eu||a.codice = i.segment2
    and a.nb_titolo = w_corr_cap;
    exception when no_data_found then totale := 0;
  end;
  end if;
  
  return(totale);
  
END;
