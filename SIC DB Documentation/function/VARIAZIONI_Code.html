function variazioni(p_anno in NUMBER, p_tipo in varchar2,p_eu in varchar2,p_capitolo in varchar2,p_segment9 in varchar2,p_tipo_atto varchar2) return number IS

-- passare il p_segment9 se serve il totale variazioni per uno specifico vincolo o FPV/AVANZO

tot_variazioni number := 0;

begin

if upper(nvl(p_tipo,'COMPE')) = 'COMPE'
 then

 -- Variazioni COMPE

     SELECT  
     nvl(sum(decode(SUBSTR(a.doc_type,7,3),'NEG',-a.document_amount,a.document_amount)),0) into  tot_variazioni
     FROM fin_t_documenti a
     WHERE a.segment8 = p_anno
     and a.doc_type in 
     (
     'V_COM_POS',
     'V_COM_NEG',
     'A_COM_POS',
     'A_COM_NEG'
     )
     and trunc(a.date_created,'dd') >= (select trunc(data_approvazione_previsione,'dd')+1 from fin_t_dati_generali where anno_finanziario = a.segment8)
     and nvl(a.deletion_status_flag,'N')<>'Y'
     and (nvl(a.segment9,'X') = p_segment9 or p_segment9 is null)
     and substr(a.segment2,1,1) = nvl(p_eu,'U')
     and (a.segment2 = p_capitolo OR p_capitolo is null) -- questa istruzione più elegante non funziona nvl(p_capitolo,a.segment2)
     and (nvl(a.attribute13,'X') = p_tipo_atto OR p_tipo_atto is null);

 else 

     -- VARIAZIONI CASSA

     SELECT  
     nvl(sum(decode(SUBSTR(a.doc_type,7,3),'NEG',-a.document_amount,a.document_amount)),0) into  tot_variazioni
     FROM fin_t_documenti a,
          fin_t_articoli b
     WHERE a.segment8 = p_anno
     and a.doc_type in 
     (
     'V_CAS_POS',
     'V_CAS_NEG',
     'A_CAS_POS',
     'A_CAS_NEG'
     )
     and trunc(a.date_created,'dd') >= (select trunc(data_approvazione_previsione,'dd')+1 from fin_t_dati_generali where anno_finanziario = a.segment8)
     and nvl(a.deletion_status_flag,'N')<>'Y'
     and (nvl(a.segment9,'X') = p_segment9 or p_segment9 is null)
     and substr(a.segment2,1,1) = nvl(p_eu,'U')
     and (a.segment2 = p_capitolo OR p_capitolo is null) -- questa istruzione più elegante non funziona nvl(p_capitolo,a.segment2)
     and (nvl(a.attribute13,'X') = p_tipo_atto OR p_tipo_atto is null);

 end if;

return(tot_variazioni);


END;
