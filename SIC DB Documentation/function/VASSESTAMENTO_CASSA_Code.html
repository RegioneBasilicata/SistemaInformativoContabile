function      vassestamento_cassa(p_anno in NUMBER, p_capitolo in varchar2, p_data in date) return number IS
importo number(15,2) := 0; 
vdata_atto_assestamento date;
vdata number;

BEGIN
vdata := to_number(to_char(p_data,'yyyymmdd'));

 begin
 select data_atto_assestamento 
 into vdata_atto_assestamento
 from fin_t_dati_generali 
 where anno_finanziario=p_anno;
 exception when no_data_found then null;
 end;

  select 
 nvl(sum(decode(substr(d.doc_type,7,3),'NEG',(d.document_amount * -1),d.document_amount)),0) 
 into importo
 from  fin_t_documenti d
 where d.segment8 = p_anno
 and d.segment2 = p_capitolo
 and to_number(to_char(d.creation_date,'yyyymmdd')) > vdata
 and 
 (
 d.doc_type in 
 ('A_CAS_POS'
 ,'A_CAS_NEG'
 ) or (doc_type in ('V_CAS_POS','V_CAS_NEG') and nvl(attribute10,'N')<>'Y') and trunc(date_created,'dd') < trunc(nvl(vdata_atto_assestamento,date_created+1),'dd') );
 
 return nvl(importo,0);
END;
