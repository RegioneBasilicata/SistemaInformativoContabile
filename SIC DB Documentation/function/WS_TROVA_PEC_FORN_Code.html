function ws_trova_pec_forn (tr_id in number) RETURN VARCHAR2  is
    l_envelope CLOB;
    l_risultato xmltype;
    l_cod_fisc varchar2(16);
    l_part_iva varchar2(11);
    tr_pec     varchar2(100);
BEGIN
select PARTITA_IVA,CODICE_FISCALE
 into l_part_iva,l_cod_fisc
 from fin_t_anagrafica
 where id=tr_id;
-- ricerca prima con ws impresa e prima con partita iva e poi con codice fiscale, poi fa' la ricerca amministrazione e poi professionista
-- restituisce la prima pec che trova 
tr_pec:=null;
if l_part_iva is not null then
  l_envelope := 
  '<soapenv:Envelope xmlns:soapenv="http://schemas.xmlsoap.org/soap/envelope/" xmlns:ns="http://www.intemaweb.com/egov/Anagrafica/AnagraficaTypes/200911">
     <soapenv:Header/>
     <soapenv:Body>
      <ns:SearchRequest pageNumber="1" pageSize="1">
        <ns:Impresa>'||l_part_iva||'</ns:Impresa>
      </ns:SearchRequest>
    </soapenv:Body>
  </soapenv:Envelope>';
  l_risultato:=apex_web_service.make_request(
                p_url => 'http://protocolloservices.regione.basilicata.it:9800/anagraficaServices', 
                p_action => 'Search',
                p_envelope => l_envelope);
  tr_pec := APEX_WEB_SERVICE.PARSE_XML(
            p_xml => l_risultato,
            p_xpath => '//Anagrafica/Pec/text()',
            p_ns => 'xmlns="http://www.intemaweb.com/egov/Anagrafica/AnagraficaTypes/200911"  xmlns:ns2="http://www.intemaweb.com/egov/Anagrafica/AnagraficaServices/200911"');
end if;
--return l_risultato.getclobval();
if tr_pec is not null then
   return tr_pec;
end if;   
if l_part_iva is not null then
  l_envelope := 
  '<soapenv:Envelope xmlns:soapenv="http://schemas.xmlsoap.org/soap/envelope/" xmlns:ns="http://www.intemaweb.com/egov/Anagrafica/AnagraficaTypes/200911">
     <soapenv:Header/>
     <soapenv:Body>
      <ns:SearchRequest pageNumber="1" pageSize="1">
			 <ns:Amministrazione>
			  	<ns:CodiceFiscale>'||l_part_iva||'</ns:CodiceFiscale>
			 </ns:Amministrazione>
      </ns:SearchRequest>
    </soapenv:Body>
  </soapenv:Envelope>';
  l_risultato:=apex_web_service.make_request(
                p_url => 'http://protocolloservices.regione.basilicata.it:9800/anagraficaServices', 
                p_action => 'Search',
                p_envelope => l_envelope);
  tr_pec := APEX_WEB_SERVICE.PARSE_XML(
            p_xml => l_risultato,
            p_xpath => '//Anagrafica/Pec/text()',
            p_ns => 'xmlns="http://www.intemaweb.com/egov/Anagrafica/AnagraficaTypes/200911"  xmlns:ns2="http://www.intemaweb.com/egov/Anagrafica/AnagraficaServices/200911"');
end if;
--return l_risultato.getclobval();
if tr_pec is not null then
   return tr_pec;
end if;   
if l_cod_fisc is not null then
  l_envelope := 
  '<soapenv:Envelope xmlns:soapenv="http://schemas.xmlsoap.org/soap/envelope/" xmlns:ns="http://www.intemaweb.com/egov/Anagrafica/AnagraficaTypes/200911">
     <soapenv:Header/>
     <soapenv:Body>
      <ns:SearchRequest pageNumber="1" pageSize="1">
			  <ns:Professionista>'||l_cod_fisc||'</ns:Professionista>
      </ns:SearchRequest>
    </soapenv:Body>
  </soapenv:Envelope>';
  l_risultato:=apex_web_service.make_request(
                p_url => 'http://protocolloservices.regione.basilicata.it:9800/anagraficaServices', 
                p_action => 'Search',
                p_envelope => l_envelope);
  tr_pec := APEX_WEB_SERVICE.PARSE_XML(
            p_xml => l_risultato,
            p_xpath => '//AnagraficaList/Anagrafica/Pec/text()',
            p_ns => 'xmlns="http://www.intemaweb.com/egov/Anagrafica/AnagraficaTypes/200911"  xmlns:ns2="http://www.intemaweb.com/egov/Anagrafica/AnagraficaServices/200911"');
end if;
--return l_risultato.getclobval();
return tr_pec;
END;
