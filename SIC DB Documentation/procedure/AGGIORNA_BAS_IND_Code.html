PROCEDURE aggiorna_bas_ind (wanno in NUMBER) IS

 capitolo_anno varchar2(6);

 cursor doc is
 select doc_id,segment2 cap
from fin_t_documenti
where doc_type in ('IMP','IMP-DEF','IMP-PER','ACC')
and segment8 = wanno - 1
and nvl(deletion_status_flag,'N') = 'N'
minus
select doc_id,ind_cap cap 
from bas_ind
where ind_anno = wanno;


BEGIN

 for d in doc loop

 begin
   select eu||codice into capitolo_anno
     from fin_t_articoli 
    where anno = wanno 
    and nvl(abilitato,'S') <> 'N' 
    and CODICE_PRECEDENTE = d.cap;
  exception when no_data_found then capitolo_anno := 'X';
  end;

 if capitolo_anno <> 'X'
   then
   begin
	    insert into bas_ind (doc_id,ind_cap,ind_anno)
      values (d.doc_id,capitolo_anno,wanno);
      commit;
   exception when others then null;
   end;

 end if;

   END LOOP; 


END;
