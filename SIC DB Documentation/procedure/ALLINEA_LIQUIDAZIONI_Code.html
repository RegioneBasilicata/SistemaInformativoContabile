procedure allinea_liquidazioni authid definer as

cursor liq is
select d.DOC_ID
,t.ID_ANA
,t.ID_SEDE
--,t.OPEN_BALANCE_PROPOSED
,open_balance_liq_tp(t.doc_id,t.id_ana,t.ID_SEDE) residuo_effettivo
from fin_t_documenti d
,fin_t_documenti_terze_parti t
where d.doc_type = 'LIQ'
and nvl(d.deletion_status_flag,'N') = 'N'
--and d.segment8 = to_char(sysdate,'yyyy')
and t.doc_id = d.DOC_ID
and t.OPEN_BALANCE_PROPOSED <> open_balance_liq_tp(t.doc_id,t.id_ana,t.ID_SEDE);

begin
 for l in liq loop
  update fin_t_documenti_terze_parti t set t.OPEN_BALANCE_PROPOSED = l.residuo_effettivo
  where t.doc_id = l.doc_id
  and t.id_ana = l.id_ana
  and t.ID_SEDE = l.id_sede;
  commit;
 end loop;
end;
