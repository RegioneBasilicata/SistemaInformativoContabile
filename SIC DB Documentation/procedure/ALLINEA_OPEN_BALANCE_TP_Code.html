procedure allinea_open_balance_tp as

residuo_acc number := 0;

-- allineo prime le uscite 
--LIQ
cursor liq is
select d.doc_id
,t.id_ana
,t.ID_SEDE
,open_balance_liq_tp(d.doc_id,t.ID_ANA,t.ID_SEDE) residuo_effettivo
from fin_t_documenti d
,fin_t_documenti_terze_parti t
where d.doc_type = 'LIQ'
and nvl(d.DELETION_STATUS_FLAG,'N') = 'N'
and t.doc_id = d.doc_id
and nvl(t.OPEN_BALANCE_PROPOSED,0) <> open_balance_liq_tp(d.doc_id,t.ID_ANA,t.ID_SEDE);

--IMP
cursor imp is
select d.doc_id
,t.id_ana
,t.ID_SEDE
,open_balance_imp_tp(d.doc_id,t.ID_ANA,t.ID_SEDE) residuo_effettivo
from fin_t_documenti d
,fin_t_documenti_terze_parti t
where d.doc_type like 'IMP%'
and nvl(d.DELETION_STATUS_FLAG,'N') = 'N'
and t.doc_id = d.doc_id
and nvl(t.OPEN_BALANCE_PROPOSED,0) <> open_balance_imp_tp(d.doc_id,t.ID_ANA,t.ID_SEDE);

-- allineo gli accertamenti
cursor accertamenti is
select d.doc_id
,t.id_ana
,t.ID_SEDE
,open_balance_acc_tp(d.doc_id,t.ID_ANA,t.ID_SEDE) residuo_effettivo
from fin_t_documenti d
,fin_t_documenti_terze_parti t
where d.doc_type = 'ACC'
and nvl(d.DELETION_STATUS_FLAG,'N') = 'N'
and t.doc_id = d.doc_id
and nvl(t.OPEN_BALANCE_PROPOSED,0) <> open_balance_acc_tp(d.doc_id,t.ID_ANA,t.ID_SEDE);


begin

 for l in liq loop
  update fin_t_documenti_terze_parti t set t.OPEN_BALANCE_PROPOSED = l.residuo_effettivo
  where t.doc_id = l.doc_id
  and t.id_ana = l.id_ana
  and t.ID_SEDE = l.id_sede;
 end loop;
 commit;

 for i in imp loop
 if i.residuo_effettivo < 0
 then 
  update fin_t_documenti_terze_parti t set t.AMOUNT = (t.AMOUNT - i.residuo_effettivo),t.OPEN_BALANCE_PROPOSED = 0
  where t.doc_id = i.doc_id
  and t.id_ana = i.id_ana
  and t.ID_SEDE = i.id_sede;
 else
  update fin_t_documenti_terze_parti t set t.OPEN_BALANCE_PROPOSED = i.residuo_effettivo
  where t.doc_id = i.doc_id
  and t.id_ana = i.id_ana
  and t.ID_SEDE = i.id_sede;
 end if;
 end loop;
 commit;

 for a in accertamenti loop
     if a.residuo_effettivo < 0
      then residuo_acc := 0;
      else residuo_acc := a.residuo_effettivo;
     end if;
 update fin_t_documenti_terze_parti t set t.OPEN_BALANCE_PROPOSED = residuo_acc
  where t.doc_id = a.doc_id
  and t.id_ana = a.id_ana
  and t.ID_SEDE = a.id_sede;
 end loop;
 commit;


end;
