procedure capitoli_precedenti (p_anno in number) is 

esiste number := 0;

cursor cap is 
select eu
,codice 
,codice_precedente
from fin_t_articoli
where anno  = p_anno
and nvl(abilitato,'N') = 'S';

begin
 for c in cap loop

   select count(*) into esiste
   from fin_t_articoli
   where anno = p_anno -1
   and eu = c.eu
   and codice = c.codice;

   if esiste > 0 --quindi se il capitolo esiste nell'anno precedente inserisco come codice precedente il capitolo stesso
    then 
    update fin_t_articoli set codice_precedente = c.eu||c.codice
    where anno = p_anno 
    and eu = c.eu
    and codice = c.codice
    and codice_precedente <> c.eu||c.codice;
    commit;
  end if;


 end loop;
end;
