procedure collega_capitoli_perenti (p_anno in number) is

cursor capitoli is
select p.CAP_ORIGINARIO capitolo
,a.fonte_finanziamento fonte
,d.dipartimento_principale dipartimento
,a.nb_titolo titolo
from fin_t_capitoli_perenti p
,fin_t_articoli a
,fin_t_capitoli_dipartimenti d
where p.anno = p_anno
and p.cap_perenzioni is null
and a.anno = p.anno
and a.eu||a.codice = p.cap_originario
and d.anno = p.anno
and d.capitolo = p.cap_originario;

begin
 for c in capitoli
 loop
  if c.titolo = 1
   then
    if c.fonte like '%REG%' or c.fonte like '%MUTUI%'
     then update fin_t_capitoli_perenti set cap_perenzioni = 'U65040' where anno = p_anno and cap_originario = c.capitolo;
     else update fin_t_capitoli_perenti set cap_perenzioni = 'U65060' where anno = p_anno and cap_originario = c.capitolo;
    end if;
   else 
    if c.fonte like '%REG%' or c.fonte like '%MUTUI%'
     then 
      if c.dipartimento in ('14','19')
       then update fin_t_capitoli_perenti set cap_perenzioni = 'U65100' where anno = p_anno and cap_originario = c.capitolo;
       else update fin_t_capitoli_perenti set cap_perenzioni = 'U65080' where anno = p_anno and cap_originario = c.capitolo; 
      end if;
    else  
      if c.dipartimento in ('14','19')
        then update fin_t_capitoli_perenti set cap_perenzioni = 'U65160' where anno = p_anno and cap_originario = c.capitolo;
        else update fin_t_capitoli_perenti set cap_perenzioni = 'U65150' where anno = p_anno and cap_originario = c.capitolo; 
      end if;
    end if;  
  end if;   
 end loop;
end;
