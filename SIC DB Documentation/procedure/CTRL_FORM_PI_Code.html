PROCEDURE CTRL_FORM_PI (w_pi IN VARCHAR2, w_errore IN OUT VARCHAR2) IS
 comodo NUMBER;
 totale NUMBER := 0;
 tot1 NUMBER;
 tot2 NUMBER;
BEGIN
if length(w_pi)=11 then 
 FOR i IN 1..5 LOOP
  comodo := SUBSTR(w_pi,i*2,1)*2;
  totale := totale + TRUNC(comodo/10) + MOD(comodo,10);
  totale := totale + SUBSTR(w_pi,i*2-1,1);
 END LOOP;
 tot1 := MOD(totale,10);
 tot2 := SUBSTR(w_pi,11,1);
 tot1 := 10 - tot1;
 tot1 := MOD(tot1,10);
 IF tot1 != tot2 THEN
  w_errore := 'S';
 ELSE
  w_errore := 'N';
 END IF;
else
  w_errore := 'S';
end if;
END CTRL_FORM_PI;
