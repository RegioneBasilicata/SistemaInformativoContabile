PROCEDURE email_allarmi
AS
from_mail varchar(4000) := 'avviso@regione.basilicata.it';
descr_mail  varchar2(4000);
begin
for m in (select * from VW_ALLARMI
           where email is not null and
                data_inizio_mail<=sysdate
                and data_esecuzione is null) loop
if m.tipo='BENI' and
   m.codice1='I' then
   descr_mail:='Riferimento allarme: Bene Immobile codice '||m.codice2||chr(13)||chr(10);
elsif m.tipo='BENI' and
      m.codice2='M' then
      descr_mail:='Riferimento allarme: Bene Mobile codice '||m.codice2||chr(13)||chr(10);
elsif m.tipo='CONTRATTI' then
      descr_mail:='Riferimento allarme: Contratto codice '||m.codice1||'/'||m.codice2||chr(13)||chr(10);
end if;
descr_mail:=descr_mail||m.desc_tipo||chr(13)||chr(10);
descr_mail:=descr_mail||m.descrizione||chr(13)||chr(10)||chr(13)||chr(10);

descr_mail:=descr_mail||'P.S. Questa mail e'' inviata automaticamente dalla procedura SIC; Non rispondere a questa mail';

apex_util.set_security_group_id(p_security_group_id =>
3024427007805833);
htmldb_mail.send(
p_to => m.email,
p_from => from_mail,
p_body => descr_mail,
p_subj => 'Messaggio allarme procedura Sic');
commit;
end loop;
APEX_MAIL.PUSH_QUEUE;
end;
