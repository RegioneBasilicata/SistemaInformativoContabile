procedure interroga_anagrafica
(
p_denominazione in varchar2,
p_cf_piva in varchar2,
p_id_ana out number,
p_codice_risposta out number,
p_descrizione_risposta out varchar2
)
is
begin
p_codice_risposta := 0;
p_descrizione_risposta := 'Operazione Terminata Correttamente';

  if p_denominazione is null and p_cf_piva is null
    then p_id_ana := 0;
         p_codice_risposta := 1;
         p_descrizione_risposta := 'Denominazione e codice fiscale/ partita IVA non possono essere entrambi vuoti!';
         return;
  end if;

  begin 
  select nvl(id,0) into p_id_ana
  from fin_t_anagrafica
  where upper(replace(nvl(denominazione,cognome||nome),' ')) = upper(replace(nvl(p_denominazione,''),' '))
  or partita_iva = p_cf_piva
  or codice_fiscale = p_cf_piva;
   exception when no_data_found then p_id_ana := 0;
         p_codice_risposta := 1;
         p_descrizione_risposta := 'Anagrafica non trovata';
  end;
end;
