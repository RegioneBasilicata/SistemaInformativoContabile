PROCEDURE invia_pdp2_otp(vanno in NUMBER, vmatricola in NUMBER, vtipo_scheda IN VARCHAR2) IS
  votp varchar2(20);
  v_id_workspace NUMBER;
  wnominativo varchar2(400);
  testo_mail clob;
  wmail varchar2(1000);

BEGIN

 BEGIN
   SELECT p.otp 
   INTO votp
   FROM pdp2_otp p
   WHERE p.anno = vanno AND
         p.matricola = vmatricola AND
		 p.tipo_scheda = vtipo_scheda AND
         p.utilizzata is null AND
         p.progressivo_creazione = (select max(x.progressivo_creazione) 
		                            from pdp2_otp x 
									where x.matricola = p.matricola AND 
									      x.tipo_scheda = p.tipo_scheda AND 
										  x.anno = p.anno) ;
 EXCEPTION WHEN NO_DATA_FOUND 
   THEN votp := 'OTP inesistente';
 END;


 BEGIN


   FOR r IN (SELECT cognome||' '||nome nominativo
             FROM fin_t_pianta_organica
             WHERE matricola = vmatricola
			 ORDER BY data) LOOP
     wNominativo := r.nominativo;     
   END LOOP;

   -- se la matricola è in pdp2_ind_mailotp_prioritari prendo la mail di là, se no da fnd_user
   wmail := NULL;
   FOR r IN ((SELECT 'B' ordine, 
              indirizzo email_address
			  FROM pdp2_ind_mailotp_prioritari
			  WHERE matricola = vmatricola
			  UNION 
			  SELECT 'A' ordine, 
              email_address
			  FROM fnd_user
			  WHERE matricola = vmatricola)
			  ORDER BY 1) 

   LOOP
     wmail := r.email_address;     
   END LOOP;

   --
   --

   IF wmail IS NOT NULL
    THEN
     SELECT workspace_id
     INTO v_id_workspace
     FROM APEX_WORKSPACES
     WHERE ROWNUM = 1;

     testo_mail := '<html>
                    <head></head>
                    <body>
                    Gentile '||wNominativo||',<br>
di seguito é riportato il tuo <b>Codice identificativo univoco personale (OTP)</b>, utilizzabile una sola volta, da apporre a titolo di validazione della tua scheda di 
valutazione personale per il '||vanno||', finalizzata alla erogazione della produttività, dopo il colloquio con il proprio valutatore e la presa visione della stessa.<br><br>
Questo codice <b>é stato generato il giorno '||to_char(sysdate,'dd/mm/yyyy')||' alle ore '||to_char(sysdate,'hh24:mi')||'</b>. Eventuali altri codici generati e comunicati precedentemente, non utilizzati, non sono più validi.<br><br>
Ti ricordiamo che la conservazione in sicurezza del presente codice é tua responsabilità  e solleva la Regione Basilicata da qualsiasi uso improprio dello stesso.<br><br> 
<b>CODICE PERSONALE: '||votp||'</b><br><br> 
<em>Tale modalità  di validazione delle schede di valutazione avviene in conformità  a quanto stabilito dagli artt. 21 e 64 -septies del  nuovo Codice  dell''amministrazione Digitale 
D.lgs 82/2005 ed in particolare "2 -septies . Un atto giuridico può essere posto in essere da un soggetto identifi cato mediante SPID, nell''ambito di un sistema informatico 
avente i requisiti fissati nelle regole tecniche adottate ai sensi dell''articolo 71, attraverso processi idonei a garantire, in maniera manifesta e inequivoca, 
l''acquisizione della sua volontà . Restano ferme le disposizioni concernenti il deposito degli atti e dei documenti in via telematica secondo la normativa anche regolamentare 
in materia di processo telematico." tramite l''utilizzo del sistema regionale di autenticazione (Smartcard e/o PIN) che rispetta le specifiche tecniche adottate da SPID in 
tema di certificazione e validazione delle Identità  digitali.</em><br><br><br>
Cordiali saluti<br>
'||to_char(sysdate,'dd/mm/yyyy')||' '||to_char(sysdate,'hh24:mi')||'<br><br>
________________________________________<br>
Non rispondere a questa mail in quanto è stata inviata attraverso un sistema automatico.
                    </body>';


     APEX_UTIL.SET_SECURITY_GROUP_ID(P_SECURITY_GROUP_ID => v_id_workspace);
     APEX_MAIL.SEND(P_TO => wmail,
                    P_FROM => 'noreply@regione.basilicata.it',
                    p_body     => testo_mail,
                    p_body_html => testo_mail,
                    P_SUBJ => 'Generazione codice OTP. Tipo Scheda: '||vtipo_scheda);
     APEX_MAIL.PUSH_QUEUE;


   END IF;
   UPDATE pdp2_otp 
   SET mail_inviata = wmail 
   WHERE matricola = vmatricola AND 
         tipo_scheda = vtipo_scheda AND
         progressivo_creazione = (SELECT MAX(x.progressivo_creazione) 
                                  FROM pdp2_otp x 
                                  WHERE x.matricola = vmatricola AND 
								        x.tipo_scheda = vtipo_scheda AND
                                        x.anno = vanno);
   COMMIT;
 EXCEPTION
   WHEN NO_DATA_FOUND THEN NULL;
 END;

END;
