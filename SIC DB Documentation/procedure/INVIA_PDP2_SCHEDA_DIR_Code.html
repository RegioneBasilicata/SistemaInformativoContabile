PROCEDURE invia_pdp2_scheda_dir(vanno IN NUMBER, vperiodo IN NUMBER) IS

 vlink VARCHAR2(2000);
 v_id_workspace NUMBER;
 wnominativo VARCHAR2(400);
 testo_mail CLOB;
 wmail VARCHAR2(1000);
 vcont NUMBER;

BEGIN

 vcont := 0;

 SELECT workspace_id
 INTO v_id_workspace
 FROM APEX_WORKSPACES;

 APEX_UTIL.SET_SECURITY_GROUP_ID(P_SECURITY_GROUP_ID => v_id_workspace);

 FOR r IN (SELECT matricola,
                  ufficio,
                  tipo_scheda
           FROM pdp2_stampa_scheda
           WHERE anno = vanno AND
                 periodo = vperiodo AND 
                 mail_inviata IS NULL AND
				 tipo_scheda = 'DIR_DG'
		   ORDER BY matricola,
		            ufficio) LOOP

   wnominativo := NULL;

   FOR s IN (SELECT cognome||' '||nome nominativo
             FROM fin_t_pianta_organica
             WHERE matricola = r.matricola
			 ORDER BY data) LOOP
     wnominativo := s.nominativo;     
   END LOOP;


   wmail := NULL;

   -- se la matricola è in pdp2_ind_mailotp_prioritari prendo la mail di là, se no da fnd_user
   FOR s IN ((SELECT 'B' ordine, 
                     indirizzo email_address
			  FROM pdp2_ind_mailotp_prioritari
			  WHERE matricola = r.matricola
			  UNION 
			  SELECT 'A' ordine, 
                     email_address
			  FROM fnd_user
			  WHERE matricola = r.matricola
			  )
			  ORDER BY 1) LOOP
     wmail := s.email_address;     
   END LOOP;

   IF wmail IS NOT NULL
   THEN

     SELECT link_file 
	 INTO vlink
     FROM pdp2_stampa_scheda
     WHERE anno = vanno AND
           periodo = vperiodo AND
           matricola = r.matricola AND 
		   ufficio = r.ufficio AND
		   tipo_scheda = r.tipo_scheda;

     testo_mail := '<html>
                    <head></head>
                    <body>
                    Gentile '||wNominativo||',<br>
al seguente link<br><br>'||vlink||'<br><br> è disponibile <b>la tua scheda di valutazione personale per il </b>'||vanno||'. <br><br><br><br>
Cordiali saluti<br>
'||to_char(sysdate,'dd/mm/yyyy')||' '||to_char(sysdate,'hh24:mi')||'<br><br>
________________________________________<br>
Non rispondere a questa mail in quanto è stata inviata attraverso un sistema automatico.
                    </body>';




      APEX_MAIL.SEND(P_TO => wmail,
                    P_FROM => 'noreply@regione.basilicata.it',
                    p_body     => testo_mail,
                    p_body_html => testo_mail,
					P_SUBJ => 'Processo di valutazione '||vanno||' – disponibile la scheda di valutazione personale '||'(scheda '||r.tipo_scheda||'/'||r.ufficio||').');
      APEX_MAIL.PUSH_QUEUE;

      vcont := vcont + 1;

      -- metto un ritardo di 1 minuto ogni 10 invii
      IF vcont = 10
        THEN
          DBMS_LOCK.sleep (60);
          vcont := 0;
      END IF;


  END IF;

  UPDATE pdp2_stampa_scheda 
  SET mail_inviata = wmail 
  WHERE matricola = r.matricola AND 
        ufficio = r.ufficio AND
        tipo_scheda = r.tipo_scheda AND
	    anno = vanno AND 
        periodo = vperiodo;

  COMMIT;

 END LOOP; -- matricola/ufficio/tipo_scheda

END;
