PROCEDURE leggi_atti_archiviati AS
    v_anno NUMBER(4);
    v_da_anno NUMBER(4);
    v_a_anno NUMBER(4);
    l_envelope CLOB;
    l_xml XMLTYPE;
    l_res clob;
    l_count_xml XMLTYPE;
    l_count number;
    v_data_atto VARCHAR2(10);
    v_numero_definitivo VARCHAR2(30);
    v_oggetto VARCHAR2(4000);
    v_cod_desc_uff_prop VARCHAR2(4000);
    v_pagina_richiesta NUMBER;
    
BEGIN

 -- leggo leggo gli atti riguardanti l'anno di sistema e l’anno precedente
 
 v_a_anno := TO_CHAR(SYSDATE,'YYYY');
 v_da_anno := v_a_anno - 1;
 
 -- per ogni anno, chiedo una "pagina" di 10 atti per volta, aumentando nel loop di 1 la pagina;
 -- dal loop sulle pagine esco quando il count degli atti contenuti nella n-sima pagina è 0;
 -- IL COMMIT LO FACCIO UNA SOLA VOLTA PER ANNO, alla fine del blocco,
 -- per evitare che un eventuale crash della procedura o del sistema dopo la cancellazione
 -- lasci la tabella senza righe o con poche righe.

 
 FOR v_anno IN v_da_anno .. v_a_anno LOOP
  
  DELETE at_atti_archiviati
  WHERE anno = v_anno;

  v_pagina_richiesta := 1;
  
  LOOP /* sulle pagine */

   l_envelope := '<soapenv:Envelope xmlns:soapenv="http://schemas.xmlsoap.org/soap/envelope/" xmlns:typ="http://www.intemaweb.com/provvedimentiamministrativi/types">
        <soapenv:Header/>
        <soapenv:Body>
          <typ:RicercaAttiApprovati>
             <Richiesta>
                <Anno>'||TO_CHAR(v_anno)||'</Anno>
                <NumeroAtto>
                   <NumeroA>'||99999||'</NumeroA>
                   <NumeroDa>'||1||'</NumeroDa>
                </NumeroAtto>
                <Paginazione>
                   <PaginaRichiesta>'||TO_CHAR(v_pagina_richiesta)||'</PaginaRichiesta>
                   <TotalePerPagina>10</TotalePerPagina>
                </Paginazione>
             </Richiesta>
             <Autenticazione>
                <Cod_Fiscale>NTMSRL96A01G942B</Cod_Fiscale>
                <Cod_Ufficio>A970001</Cod_Ufficio>
             </Autenticazione>
          </typ:RicercaAttiApprovati>
        </soapenv:Body>
      </soapenv:Envelope>';

   l_xml := APEX_WEB_SERVICE.MAKE_REQUEST(
      p_url               => 'http://172.18.17.91/WSProvvedimentiFinder/ServizioRicerca.svc',
      p_action            => 'urn:RicercaAttiApprovati',
      p_envelope          => l_envelope);


   -- conto atti nell'xml ricevuto
   SELECT XMLQUERY('count(//Atto)' PASSING l_xml returning content) 
   INTO l_count_xml 
   FROM DUAL;

   l_count:= l_count_xml.getnumberval;

   -- per ogni atto estraggo alcuni attributi

   FOR i in 1..l_count LOOP /* sugli atti della pagina */

     l_res := APEX_WEB_SERVICE.PARSE_XML(
     p_xml => l_xml,
     p_xpath => '//Atto['||i||']/DataAtto/text()');
     v_data_atto := SUBSTR(l_res,9,2)||'/'||SUBSTR(l_res,6,2)||'/'||SUBSTR(l_res,1,4);
     l_res := APEX_WEB_SERVICE.PARSE_XML(
     p_xml => l_xml,
     p_xpath => '//Atto['||i||']/NumeroDefinitivo/text()');
     v_numero_definitivo := l_res;

     l_res := APEX_WEB_SERVICE.PARSE_XML(
     p_xml => l_xml,
     p_xpath => '//Atto['||i||']/Oggetto/text()');
     v_oggetto := SUBSTR(l_res,1,4000);

     l_res := APEX_WEB_SERVICE.PARSE_XML(
     p_xml => l_xml,
     p_xpath => '//Atto['||i||']/UfficioProponente/DescrizioneUfficio/text()');
     v_cod_desc_uff_prop := l_res;

     l_res := APEX_WEB_SERVICE.PARSE_XML(
     p_xml => l_xml,
     p_xpath => '//Atto['||i||']/UfficioProponente/CodiceUfficio/text()');
     v_cod_desc_uff_prop := SUBSTR(v_cod_desc_uff_prop||' ['||l_res||']',1,4000);
 
     INSERT INTO at_atti_archiviati(anno,numero_definitivo,oggetto,cod_desc_uff_prop) 
     VALUES(v_anno,v_numero_definitivo,v_oggetto,v_cod_desc_uff_prop);
     -- faccio il commit ogni 100 pagine = 1000 atti
     IF MOD(v_pagina_richiesta,100) = 0 
       THEN 
        COMMIT; 
     END IF;

   END LOOP; /* sugli atti della pagina*/
   
   IF l_count = 0 
    THEN 
      EXIT;
     ELSE 
      v_pagina_richiesta := v_pagina_richiesta + 1;   
   END IF;
  END LOOP; /* sulle pagine */

  COMMIT;
  
 END LOOP; /* sugli anni */
 setta_data_def_atti;
END;
