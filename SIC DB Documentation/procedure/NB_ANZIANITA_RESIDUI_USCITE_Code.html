procedure nb_anzianita_residui_uscite (p_anno in number) is

p_res_anno1 number(15,2);
p_res_anno2 number(15,2);
p_res_anno3 number(15,2);
p_res_anno4 number(15,2);
p_res_anno5 number(15,2);
p_res_anno6 number(15,2);
p_res_anno7 number(15,2);
p_rimanenza number(15,2);
p_res_correnti number(15,2);

p_residuo_per number(15,2);
eco_res_non_perente number(15,2);
mand_res_non_perente number(15,2);


cursor residui is
select d.segment8 anno
,b.ind_cap capitolo
,d.doc_type tipo_doc
,d.doc_sequence_value numero_doc
,nvl(residuo_imp_data(d.doc_id,to_date('3112'||p_anno,'ddmmyyyy')),0) residuo
from fin_t_documenti d
,bas_ind b
where d.doc_type like 'IMP%'
and nvl(d.deletion_status_flag,'N') = 'N'
and upper(nvl(d.proposal_status_flag,'Z'))||upper(nvl(d.auditing_status_flag,'Z'))='PI' 
and d.segment8 <= p_anno
and nvl(residuo_imp_data(d.doc_id,to_date('3112'||p_anno,'ddmmyyyy')),0) > 0
and b.ind_anno = p_anno
and b.doc_id = d.doc_id;

cursor perenti is 
select d.segment8 anno
,b.ind_cap capitolo
,d.doc_type tipo_doc
,d.doc_sequence_value numero_doc
,d.doc_id doc_id
,nvl(residuo_imp_data(d.doc_id,to_date('3112'||p_anno-1,'ddmmyyyy')),0) - nvl(d.closed_amount_proposed,0) residuo
from fin_t_documenti d
,bas_ind b
where d.doc_type like 'IMP%'
and nvl(d.deletion_status_flag,'N') = 'N'
and upper(nvl(d.proposal_status_flag,'Z'))||upper(nvl(d.auditing_status_flag,'Z'))='PT' 
and d.segment8 <= p_anno
and nvl(residuo_imp_data(d.doc_id,to_date('3112'||p_anno-1,'ddmmyyyy')),0) - nvl(d.closed_amount_proposed,0) > 0 ---occorre mettere questo anche nella parte select e togliere eventuali eco e mand
and d.data_perenzione is not null
and to_char(data_perenzione,'yyyy') > 2013
and to_char(data_perenzione,'yyyy') <= p_anno
and b.ind_anno = p_anno
and b.doc_id = d.doc_id;

cursor perenti_att is 
select d.segment8 anno
,b.ind_cap capitolo
,d.doc_type tipo_doc
,d.doc_sequence_value numero_doc
,d.doc_id doc_id
,nvl(residuo_imp_data(d.doc_id,to_date('3112'||p_anno,'ddmmyyyy')),0) - nvl(d.closed_amount_proposed,0) residuo
from fin_t_documenti d
,bas_ind b
where d.doc_type like 'IMP%'
and nvl(d.deletion_status_flag,'N') = 'N'
and upper(nvl(d.proposal_status_flag,'Z'))||upper(nvl(d.auditing_status_flag,'Z'))='PT' 
and d.segment8 <= p_anno
and nvl(residuo_imp_data(d.doc_id,to_date('3112'||p_anno,'ddmmyyyy')),0) - nvl(d.closed_amount_proposed,0) > 0 ---occorre mettere questo anche nella parte select e togliere eventuali eco e mand
and d.data_perenzione is not null
and to_char(data_perenzione,'yyyy') > p_anno
and b.ind_anno = p_anno
and b.doc_id = d.doc_id;

begin 
 delete nb_anzianita_residui where anno = p_anno and substr(capitolo,1,1) = 'U' ;
 
 
 for r in residui loop
    if r.anno <= p_anno - 7 
       then p_res_anno1 := nvl(r.residuo,0);
    end if;  
      if r.anno = p_anno - 6 
       then p_res_anno2 := nvl(r.residuo,0);
      end if;
        if r.anno = p_anno - 5 
         then p_res_anno3 := nvl(r.residuo,0);
        end if;
          if r.anno = p_anno - 4 
           then p_res_anno4 := nvl(r.residuo,0);
          end if;
             if r.anno = p_anno - 3
              then p_res_anno5 := nvl(r.residuo,0);
             end if;
               if r.anno = p_anno - 2
                then p_res_anno6 := nvl(r.residuo,0);
               end if;
                 if r.anno = p_anno - 1
                  then p_res_anno7 := nvl(r.residuo,0);
                 end if;
                   if r.anno = p_anno 
                    then p_res_correnti := nvl(r.residuo,0);
                   end if;
    p_rimanenza := nvl(p_res_anno1,0) + nvl(p_res_anno2,0) + nvl(p_res_anno3,0) + nvl(p_res_anno4,0) + nvl(p_res_anno5,0) + nvl(p_res_anno6,0) + nvl(p_res_anno7,0); 

    insert into nb_anzianita_residui (ANNO
      ,CAPITOLO
      ,TIPO_DOC
      ,NUMERO_DOC
      ,RES_ANNO1
      ,RES_ANNO2
      ,RES_ANNO3
      ,RES_ANNO4
      ,RES_ANNO5
      ,RES_ANNO6
      ,RES_ANNO7
      ,RIMANENZA
      ,RES_CORRENTI) 
      values(p_anno
      ,r.capitolo
      ,r.tipo_doc
      ,r.numero_doc
      ,nvl(p_res_anno1,0)
      ,nvl(p_res_anno2,0)
      ,nvl(p_res_anno3,0)
      ,nvl(p_res_anno4,0)
      ,nvl(p_res_anno5,0)
      ,nvl(p_res_anno6,0)
      ,nvl(p_res_anno7,0)
      ,nvl(p_rimanenza,0)
      ,nvl(p_res_correnti,0));
      commit;
      
       p_res_anno1 := 0;
       p_res_anno2 := 0;
       p_res_anno3 := 0;
       p_res_anno4 := 0;
       p_res_anno5 := 0;
       p_res_anno6 := 0;
       p_res_anno7 := 0;
       p_rimanenza := 0;
       p_res_correnti:= 0;
    end loop;
    
     -- **************************************************************************************************************************************************** 
     -- ************ correttivo per tenere conto della ECO su imp 201200253 fatta nel 2014 nonstante l'imp fosse stato mandato tutto in perenzione nell'anno
      if p_anno = 2014
       then 
       insert into nb_anzianita_residui (ANNO
      ,CAPITOLO
      ,TIPO_DOC
      ,NUMERO_DOC
      ,RES_ANNO1
      ,RES_ANNO2
      ,RES_ANNO3
      ,RES_ANNO4
      ,RES_ANNO5
      ,RES_ANNO6
      ,RES_ANNO7
      ,RIMANENZA
      ,RES_CORRENTI) 
      values(p_anno
      ,'U39050'
      ,'IMP'
      ,201200253
      ,0           
      ,0           
      ,0           
      ,0           
      ,0           
      ,-45360.33   
      ,0           
      ,-45360.33   
      ,0);
      commit;
      end if;
    -- ************ FINE CORRETTIVO
    -- **************************************************************************************************
    
-- gestione perenti parziali anni prec
 for x in perenti loop
    
    select nvl(sum(nvl(open_balance_proposed,0)),0) into eco_res_non_perente
    from fin_t_documenti
    where doc_type = 'ECO'
    and nvl(deletion_status_flag,'N') = 'N'
    and previous_doc_id = x.doc_id
    and segment8 = p_anno;
  
    select nvl(sum(nvl(open_balance_proposed,0)),0) into mand_res_non_perente
    from fin_t_documenti
    where doc_type = 'MAND'
    and nvl(deletion_status_flag,'N') = 'N'
    and doc_id_capo_filiera = x.doc_id
    and segment8 = p_anno;
   
      if (x.residuo - nvl(mand_res_non_perente,0)) >= eco_res_non_perente
        then eco_res_non_perente := nvl(eco_res_non_perente,0);
        else eco_res_non_perente := 0;
      end if;
   p_residuo_per := (nvl(x.residuo,0) - nvl(mand_res_non_perente,0)) - nvl(eco_res_non_perente,0);
 
   if  p_residuo_per > 0
    then  if x.anno <= p_anno - 7 
           then p_res_anno1 := nvl(p_residuo_per,0);
            elsif x.anno = p_anno - 6 
             then p_res_anno2 := nvl(p_residuo_per,0);
              elsif x.anno = p_anno - 5 
               then p_res_anno3 := nvl(p_residuo_per,0);
                elsif x.anno = p_anno - 4 
                 then p_res_anno4 := nvl(p_residuo_per,0);
                  elsif x.anno = p_anno - 3
                   then p_res_anno5 := nvl(p_residuo_per,0);
                    elsif x.anno = p_anno - 2
                      then p_res_anno6 := nvl(p_residuo_per,0);
                       elsif x.anno = p_anno - 1
                        then p_res_anno7 := nvl(p_residuo_per,0);
                         elsif x.anno = p_anno 
                          then p_res_correnti := nvl(p_residuo_per,0);
            end if;
    p_rimanenza := nvl(p_res_anno1,0) + nvl(p_res_anno2,0) + nvl(p_res_anno3,0) + nvl(p_res_anno4,0) + nvl(p_res_anno5,0) + nvl(p_res_anno6,0) + nvl(p_res_anno7,0); 
    
   insert into nb_anzianita_residui (ANNO
      ,CAPITOLO
      ,TIPO_DOC
      ,NUMERO_DOC
      ,RES_ANNO1
      ,RES_ANNO2
      ,RES_ANNO3
      ,RES_ANNO4
      ,RES_ANNO5
      ,RES_ANNO6
      ,RES_ANNO7
      ,RIMANENZA
      ,RES_CORRENTI) 
      values(p_anno
      ,x.capitolo
      ,x.tipo_doc
      ,x.numero_doc
      ,nvl(p_res_anno1,0)
      ,nvl(p_res_anno2,0)
      ,nvl(p_res_anno3,0)
      ,nvl(p_res_anno4,0)
      ,nvl(p_res_anno5,0)
      ,nvl(p_res_anno6,0)
      ,nvl(p_res_anno7,0)
      ,nvl(p_rimanenza,0)
      ,nvl(p_res_correnti,0));
      commit;
      p_res_anno1 := 0;
       p_res_anno2 := 0;
       p_res_anno3 := 0;
       p_res_anno4 := 0;
       p_res_anno5 := 0;
       p_res_anno6 := 0;
       p_res_anno7 := 0;
       p_rimanenza := 0;
       p_res_correnti:= 0;
       eco_res_non_perente := 0;
       mand_res_non_perente := 0;
       p_residuo_per := 0;
     end if;
    
    end loop;
-- ********** FINE gestione perenti parz. anni prec

-- gestione perenzione parz. attuale
  for r in perenti_att loop
    if r.anno <= p_anno - 7 
       then p_res_anno1 := nvl(r.residuo,0);
    end if;  
      if r.anno = p_anno - 6 
       then p_res_anno2 := nvl(r.residuo,0);
      end if;
        if r.anno = p_anno - 5 
         then p_res_anno3 := nvl(r.residuo,0);
        end if;
          if r.anno = p_anno - 4 
           then p_res_anno4 := nvl(r.residuo,0);
          end if;
             if r.anno = p_anno - 3
              then p_res_anno5 := nvl(r.residuo,0);
             end if;
               if r.anno = p_anno - 2
                then p_res_anno6 := nvl(r.residuo,0);
               end if;
                 if r.anno = p_anno - 1
                  then p_res_anno7 := nvl(r.residuo,0);
                 end if;
                   if r.anno = p_anno 
                    then p_res_correnti := nvl(r.residuo,0);
                   end if;
    p_rimanenza := nvl(p_res_anno1,0) + nvl(p_res_anno2,0) + nvl(p_res_anno3,0) + nvl(p_res_anno4,0) + nvl(p_res_anno5,0) + nvl(p_res_anno6,0) + nvl(p_res_anno7,0); 

    insert into nb_anzianita_residui (ANNO
      ,CAPITOLO
      ,TIPO_DOC
      ,NUMERO_DOC
      ,RES_ANNO1
      ,RES_ANNO2
      ,RES_ANNO3
      ,RES_ANNO4
      ,RES_ANNO5
      ,RES_ANNO6
      ,RES_ANNO7
      ,RIMANENZA
      ,RES_CORRENTI) 
      values(p_anno
      ,r.capitolo
      ,r.tipo_doc
      ,r.numero_doc
      ,nvl(p_res_anno1,0)
      ,nvl(p_res_anno2,0)
      ,nvl(p_res_anno3,0)
      ,nvl(p_res_anno4,0)
      ,nvl(p_res_anno5,0)
      ,nvl(p_res_anno6,0)
      ,nvl(p_res_anno7,0)
      ,nvl(p_rimanenza,0)
      ,nvl(p_res_correnti,0));
      commit;
       
       p_res_anno1 := 0;
       p_res_anno2 := 0;
       p_res_anno3 := 0;
       p_res_anno4 := 0;
       p_res_anno5 := 0;
       p_res_anno6 := 0;
       p_res_anno7 := 0;
       p_rimanenza := 0;
       p_res_correnti:= 0;
    end loop;
end;
