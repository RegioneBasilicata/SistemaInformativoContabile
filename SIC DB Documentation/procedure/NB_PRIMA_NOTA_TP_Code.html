procedure nb_prima_nota_tp (p_anno number) is

begin

delete nb_prima_nota_corpo_tp where to_char(t_data,'yyyy') = p_anno;
commit;

insert into nb_prima_nota_corpo_tp (T_DATA
,T_NUM_REG
,CONTO
,PROGR
,DOC_ID
,ID_ANA
,IMPORTO
,DENOMINAZIONE_ANA)
select c.t_data
,c.T_NUM_REG
,c.CONTO
,c.PROGR
,t.doc_id
,p.id_ana
,p.amount
,nvl(a.denominazione,a.nome||' '||a.cognome) denominazione_ana
from nb_prima_nota_testa t
,nb_prima_nota_corpo c
,fin_t_documenti_terze_parti p
,fin_t_anagrafica a
where to_char(t.data,'yyyy') = p_anno
and t.doc_id is not null
and c.t_data = t.data
and c.t_num_reg = t.num_reg
and p.doc_id = t.doc_id
and a.id = p.id_ana;
commit;

insert into nb_prima_nota_corpo_tp (T_DATA
,T_NUM_REG
,CONTO
,PROGR
,DOC_ID
,ID_ANA
,IMPORTO
,DENOMINAZIONE_ANA)
select c.t_data
,c.T_NUM_REG
,c.CONTO
,c.PROGR
,null
,c.id_cred_deb
,c.importo
,nvl(a.denominazione,a.nome||' '||a.cognome) denominazione_ana
from nb_prima_nota_testa t
,nb_prima_nota_corpo c
,fin_t_anagrafica a
where to_char(t.data,'yyyy') = p_anno
and t.doc_id is null
and c.t_data = t.data
and c.t_num_reg = t.num_reg
and c.id_cred_deb is not null
and a.id = c.id_cred_deb;
commit;

end;
