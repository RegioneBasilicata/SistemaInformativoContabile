PROCEDURE PuliziaPiantaOrganica is
begin
for d in (SELECT MIN(rowid) riga,trunc(DATA,'DD') data1,MATRICOLA,COGNOME,NOME,CODFISC,CODAREA,CODUFF,CODSERV,CCODPRESIDIO,DESCRAREA,
DESCRUFFICIO,DESCRSERV,DESCRPRESIDIO,DATAASS,LIVELLO,CODQUA,QUALIFICA,CODPOSORG,POSIZIONEORG,
NUMTEL,DATACESS,EMAIL,count(*)
FROM fin_t_pianta_organica b
GROUP BY trunc(DATA,'DD'),MATRICOLA,COGNOME,NOME,CODFISC,CODAREA,CODUFF,CODSERV,CCODPRESIDIO,DESCRAREA,
DESCRUFFICIO,DESCRSERV,DESCRPRESIDIO,DATAASS,LIVELLO,CODQUA,QUALIFICA,CODPOSORG,POSIZIONEORG,
NUMTEL,DATACESS,EMAIL
having count(*)>1) loop
delete fin_t_pianta_organica b 
where 
trunc(b.DATA,'dd')=d.data1 and
b.MATRICOLA=d.matricola and
b.COGNOME=d.cognome and
b.NOME=d.nome and
b.CODFISC=d.codfisc and
nvl(b.CODAREA,' ')=nvl(d.codarea,' ') and
nvl(b.CODUFF,' ')=nvl(d.coduff,' ') and
nvl(b.CODSERV,' ')=nvl(d.codserv,' ') and
nvl(b.CCODPRESIDIO,' ')=nvl(d.ccodpresidio,' ') and
nvl(b.DESCRAREA,' ')=nvl(d.descrarea,' ') and
nvl(b.DESCRUFFICIO,' ')=nvl(d.descrufficio,' ') and
nvl(b.DESCRSERV,' ')=nvl(d.descrserv,' ') and
nvl(b.DESCRPRESIDIO,' ')=nvl(d.descrpresidio,' ') and
trunc(b.DATAASS,'dd')=trunc(d.dataass,'dd') and
nvl(b.LIVELLO,' ')=nvl(d.livello,' ') and
nvl(b.CODQUA,' ')=nvl(d.codqua,' ') and
nvl(b.QUALIFICA,' ')=nvl(d.qualifica,' ') and
nvl(b.CODPOSORG,' ')=nvl(d.codposorg,' ') and
nvl(b.POSIZIONEORG,' ')=nvl(d.posizioneorg,' ') and
nvl(b.NUMTEL,' ')=nvl(d.numtel,' ') and
nvl(b.DATACESS,to_date('01010001','ddmmyyyy'))=nvl(d.datacess,to_date('01010001','ddmmyyyy')) and
nvl(b.EMAIL,' ')=nvl(d.email,' ') and
b.rowid<>d.riga;

end loop;
commit;
end;
