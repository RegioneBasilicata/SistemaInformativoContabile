procedure quadri as
  id_doc      number;
  progr       number;
  errore      number;
  wdocumento varchar2(200);
  in_file UTL_FILE.FILE_TYPE;
  linebuf  VARCHAR2(11);
  a_cod_inv varchar2(100);
BEGIN
   --CREATE DIRECTORY DIR_QUADRI AS '/file_repository/file_esterni/quadri';
   in_file:=UTL_FILE.FOPEN('DIR_QUADRI','quadri.txt','r');
 LOOP
  UTL_FILE.GET_LINE(in_file,linebuf);
  linebuf:=substr(linebuf,1,10);	
  errore :=null;
  begin
   select cod_inv
     into a_cod_inv
     from tb_beni_mobili
     where num_inv=substr(linebuf,1,6);	
  exception
    WHEN no_data_found THEN
	  DBMS_OUTPUT.PUT_LINE (linebuf); 
      errore:=1;       
  end;   
  if errore is null then
    select doc_seq.nextval
      into id_doc
      from dual;

    wdocumento := id_doc||linebuf;
    UTL_FILE.FRENAME('DIR_QUADRI', linebuf , 'DIR_QUADRI', wdocumento, TRUE);

    insert into fina_file_esterni
      (id,nome_file)
      values
      (id_doc,linebuf);
    SELECT nvl(max(progr),0) + 1
        into progr
        FROM  tb_doc_cesp
        WHERE codice=a_cod_inv;
    insert into tb_doc_cesp
      (CODICE,PROGR,DOCUMENTO,DESCRIZIONE,TIPO,CREATO_DA,CREATO_IL)
       values
      (a_cod_inv,progr,id_doc,'Foto','M',1000,sysdate);
  end if;
 END LOOP;
 UTL_FILE.FCLOSE(in_file);
 --DROP DIRECTORY DIR_QUADRI
EXCEPTION
 WHEN no_data_found THEN
    UTL_FILE.FCLOSE(in_file);
      COMMIT;
	 -- DROP DIRECTORY DIR_QUADRI
END;
