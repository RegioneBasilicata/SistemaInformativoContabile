procedure scompatta_lettura_mandati (p_dal in date, p_al in date) as

 cursor mandati is
 select 
 r.rev_mand_cod 		  numero_mandato,
 m.importo_totale_mandato importo_totale_mandato,
 m.data_emissione_mandato data_emissione_mandato,
 m.codice_sede_fornitore  codice_sede_fornitore,
 m.id_fornitore 		  id_fornitore,
 r.rev_codice 			  reversale_collegata
 from		
 fina_rev_coll r, 
 fina_mandati m 
 where  
 r.rev_mand_cod = m.numero_mandato AND
 M.data_emissione_mandato between p_dal and p_al; 

 
 r mandati%ROWTYPE;
BEGIN

-- azzero la tabella di appoggio per l'utente che esegue la stampa 
delete fina_reversali_collegate where utente = nvl(v('APP_USER'),user);

commit;
 FOR r IN mandati LOOP

  insert into fina_reversali_collegate
   (UTENTE, NUMERO_MANDATO, IMPORTO_MANDATO, 
    CODICE_SEDE_BENEFICIARIO, ID_BENEFICIARIO
    REVERSALE_COLLEGATA, IMPORTO_REVERSALE_COLLEGATA)
   values 
   (nvl(v('APP_USER'),user), r.numero_mandato,
    r.importo_totale_mandato, r.codice_sede_fornitore,
  	r.id_fornitore, r.reversale_collegata, null);  
 
 END LOOP;
commit;
END scompatta_lettura_mandati;
