PROCEDURE setta_data_def_atti as
    l_envelope CLOB;
    l_xml XMLTYPE;
    l_count_step_xml XMLTYPE;
    l_count_step number;
    l_data_arr varchar2(4000);
    l_utente varchar2(4000);

    intema            varchar2(1);
    l_num_atto        varchar2(20);
    l_numero          varchar2(100);
BEGIN
 FOR atto IN (select numero_definitivo from at_atti_archiviati where anno between TO_CHAR(SYSDATE,'YYYY')-1 and  TO_CHAR(SYSDATE,'YYYY') and data_archivio is null) LOOP 

    l_envelope := '<soapenv:Envelope xmlns:soapenv="http://schemas.xmlsoap.org/soap/envelope/" xmlns:typ="http://www.intemaweb.com/provvedimentiamministrativi/types">
        <soapenv:Header/>
        <soapenv:Body>
          <typ:GetStorico>
             <Richiesta>
               <NumeroDefinitivo>'||atto.numero_definitivo||'</NumeroDefinitivo>
             </Richiesta>
             <Autenticazione>
                <Cod_Fiscale>NTMSRL96A01G942B</Cod_Fiscale>
                <Cod_Ufficio>A970001</Cod_Ufficio>
             </Autenticazione>
          </typ:GetStorico>
        </soapenv:Body>
      </soapenv:Envelope>';

   l_xml := APEX_WEB_SERVICE.MAKE_REQUEST(
      p_url               => 'http://172.18.17.91/WSProvvedimentiFinderStorico/ServizioRicerca.svc',
      p_action            => 'urn:GetStorico',
      p_envelope          => l_envelope);

     -- conto allegati nell'atto
    SELECT XMLQUERY('count(//Step)' passing l_xml returning content) 
    INTO l_count_step_xml 
    FROM dual;
    l_count_step:= l_count_step_xml.getnumberval;

    FOR k in 1..l_count_step LOOP
      l_data_arr := apex_web_service.parse_xml(p_xml => l_xml, p_xpath => '//Step['||k||']/DataArrivo/text()');
      l_utente := apex_web_service.parse_xml(p_xml => l_xml, p_xpath => '//Step['||k||']/Utente/text()');
      if l_utente='ARCHIVIO' then
         update at_atti_archiviati
           set data_archivio=to_date(l_data_arr,'yyyy-mm-dd')
           where numero_definitivo=atto.numero_definitivo;
         commit;  
      end if;    
    END LOOP; /* step */
 end loop;   
END;
