procedure situazione_cc (w_cc in number,w_km out number,w_spesa out number) is
 da_data date;
 a_data  date;
 tot_spese  number;
 tot_km_cc   number;
 tot_km_gen   number;
 a_tipo_spese varchar2(15);
 tot_gas_gen number;
begin
select da_periodo, a_periodo, tipo_spese
   into da_data,a_data,a_tipo_spese
   from ap_centri_costo
   where codice=w_cc;
-- calcolo km
select round(sum((nvl(V.KM_FINE,V.KM_INZIO)-V.KM_INZIO)*nvl(p.percentuale,100)/100),0)
  into tot_km_cc
  from ap_viaggi v,vw_perc_cc_viaggio p
  where v.data_partenza between da_data and a_data and
        v.centro_costo=w_cc and
        nvl(v.escludi_calcolo_cc,'N')='N' and
        p.codice_viaggio(+)=v.codice and
        P.CENTRO_COSTO(+)=v.centro_costo;

-- calcolo spese viaggi generati da richieste 
if instr(a_tipo_spese,'X')<>0 and instr(a_tipo_spese,'Y')<>0 then
 select round(sum(s.importo*nvl(p.percentuale,100)/100),0)
  into tot_spese
  from ap_viaggi v,vw_perc_cc_viaggio p,vw_spese_viaggio s
  where v.data_partenza between da_data and a_data and
        v.centro_costo=w_cc and
        nvl(v.escludi_calcolo_cc,'N')='N' and
        p.codice_viaggio(+)=v.codice and 
        P.CENTRO_COSTO(+)=v.centro_costo and
        s.codice=v.codice and
        s.tipologia in a_tipo_spese;
end if;        
if instr(a_tipo_spese,'X')>0 then
-- calcolo spese rapporto benzina / km calcolato come sommatoria Totale gasolio generale : x = totale km viaggi generale :  totale km c/c
 select sum(nvl(V.KM_FINE,V.KM_INZIO)-V.KM_INZIO)
  into tot_km_gen
  from ap_viaggi v
  where v.data_partenza between da_data and a_data and
        v.centro_costo is not null and
        nvl(v.escludi_calcolo_cc,'N')='N';
 select sum(s.importo)
 into tot_gas_gen
 from ap_viaggi v,vw_spese_viaggio s
 where v.data_partenza between da_data and a_data and
       v.centro_costo is not null and
       nvl(v.escludi_calcolo_cc,'N')='N' and
       s.codice=v.codice and
       s.tipologia='G';
 tot_spese:=round((tot_gas_gen*tot_km_cc/tot_km_gen),0);
end if;
if instr(a_tipo_spese,'Y')>0 then
-- calcolo spese da tabella chilometrica
 select sum(
   case when km_fin<tot_km_cc then
     (km_fin-km_ini+1)*costo
     else
     (tot_km_cc-km_ini+1)*costo
     end ) costo
   into tot_spese
   from ap_cc_costi
   where codice=w_cc and
   km_ini<tot_km_cc;
end if;

W_KM:=nvl(tot_km_cc,0);
W_SPESA:=nvl(tot_spese,0);
end;
