procedure SP_LIQ(p_anno in number, p_data in date) authid definer as

esercizio number := p_anno;

id_causale number := 0;
descr_causale varchar2(2000);
id_liq number;

id_pn_testa number := 0;
indice number :=0; --indice
conto_dare number;
conto_avere number;
conto_coll number;
p_data_mov date := nvl(p_data,to_date('3112'||esercizio,'ddmmyyyy'));

cursor liq is
select distinct d.date_created data_doc,
d.segment2 capitolo
,d.doc_id doc_id
,d.description||' - Liquidazione n.'||d.doc_sequence_value||' - Capitolo '||d.segment2 descrizione
,nvl(d.document_amount,0) importo_liq
,a.conto_sanita
,nvl(d.LAST_UPDATED_BY,d.created_by) utente
,p.conto_partite_giro
from fin_t_documenti d
, fin_t_articoli a
, sa_piano_dei_conti p
, fin_t_documenti_terze_parti t
where d.segment8 = esercizio
and d.doc_type = 'LIQ'
and nvl(d.deletion_status_flag,'N') = 'N'
and trunc(d.date_created,'dd') <= trunc(p_data_mov,'dd')
and a.anno = d.segment8
and a.eu||a.codice = d.segment2 
and nvl(a.sanita_sn,'N') = 'S'--and nvl(a.conto_sanita,'X') <> 'X'
and t.doc_id = d.doc_id
and p.codice = a.conto_sanita
;--and t.id_ana in (32149,107093,107100,32145,32200);

cursor dettaglio is
select nvl(t.amount,0) importo
,p.codice codice_conto
,p.descrizione descr_conto
from fin_t_documenti d
, fin_t_documenti_terze_parti t
, sa_piano_dei_conti p
where d.doc_id = id_liq
and t.doc_id = d.doc_id
and p.codice_beneficiario = t.id_ana
and p.conto_collegato = conto_coll;

BEGIN

for l in liq loop
id_liq := l.doc_id;
conto_coll := nvl(l.conto_partite_giro,l.conto_sanita);

id_pn_testa := 0;
-- trovo la causale contabile relativa al capitolo
  begin
    select nvl(id,0),nvl(denominazione,'X')
      into id_causale,descr_causale
    from sa_causali_contabili
    where capitolo = 'U00000'
    and doc_type = 'LIQ';
    exception when no_data_found then id_causale:= 0; descr_causale := '';
  end;
  -- PRIMA NOTA TESTA
  -- stacca id per sa_prima_nota_testa
      select nvl(max(num_reg),0) + 1 into id_pn_testa
      from sa_prima_nota_testa
      where data = l.data_doc;
    --FINE   
    -- inserisco la testata in sa_prima_nota_testa
    insert into sa_prima_nota_testa(DATA,NUM_REG,DOC_ID,DESCRIZIONE_MOV,SA_CC_ID,utente)
    values
    (l.data_doc,
     id_pn_testa,
     l.doc_id,
     l.descrizione,
     id_causale,
     l.utente
    );
    commit;
  -- FINE PNT
  
     -- stacco identificativo univoco e progressivo per il rigo della sa_prima_nota_corpo
        SELECT sq_sapc.NEXTVAL into indice 
        FROM dual;
        -- ****
        -- inserisco scrittura per riduzione conto DEBITI GENERICI e ribaltamento sui conti di Debito delle aziende
        insert into sa_prima_nota_corpo (T_DATA,T_NUM_REG,ID_CONTO,PROGR,D_A,IMPORTO)
        values
        (l.data_doc,
        id_pn_testa,
        l.conto_sanita, -- conto DEBITI generici oppure fondo se partite di giro
         indice,
         'D',
         nvl(l.importo_liq,0)
        );
        commit;

    
    -- FINE causali guidate
  
  --- ************************
   for d in dettaglio loop
      
    -- stacco identificativo univoco e progressivo per il rigo della sa_prima_nota_corpo
    SELECT sq_sapc.NEXTVAL into indice 
    FROM dual;
    -- ****
    insert into sa_prima_nota_corpo (T_DATA,T_NUM_REG,ID_CONTO,PROGR,D_A,IMPORTO)
    values
    (l.data_doc,
    id_pn_testa,
    d.codice_conto, -- Debiti v/aziende oppure fondo se partite di giro
    indice,
    'A',
    nvl(d.importo,0)
    );
    commit;
 
  -- FINE causali guidate
-- FINE scrittura debiti vs/aziende a Tesoreria
  end loop;
    end loop;  

 
END;
