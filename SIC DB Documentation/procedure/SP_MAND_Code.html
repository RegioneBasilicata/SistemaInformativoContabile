procedure SP_MAND(p_anno in number, p_data in date) authid definer as

esercizio number := p_anno;

id_causale number := 0;
descr_causale varchar2(2000);
id_mand number;

id_pn_testa number := 0;
indice number :=0; --indice
conto_dare number;
conto_avere number;
conto_coll number;
p_data_mov date := nvl(p_data,to_date('3112'||esercizio,'ddmmyyyy'));

cursor mand is
select distinct d.date_created data_doc,
d.segment2 capitolo
,d.doc_id doc_id
,d.description||' - Mandato n.'||d.doc_sequence_value||' - Capitolo '||d.segment2 descrizione
,nvl(d.document_amount,0) importo_mand
,a.conto_sanita
,nvl(d.LAST_UPDATED_BY,d.created_by) utente
,p.conto_partite_giro
from fin_t_documenti d
, fin_t_articoli a
, sa_piano_dei_conti p
, fin_t_documenti_terze_parti t
, fin_distinte_mand_testata x
, fin_distinte_mand_dettaglio y
where d.segment8 = esercizio
and d.doc_type = 'MAND'
and nvl(d.open_balance_proposed,0) > 0
and trunc(d.date_created,'dd') <= trunc(p_data_mov,'dd')
and a.anno = d.segment8
and a.eu||a.codice = d.segment2 
and nvl(a.sanita_sn,'N') = 'S'
and p.codice = a.conto_sanita
and t.doc_id = d.doc_id
and y.ANNO_FINANZIARIO = d.segment8
and y.NUMERO_MANDATO = d.doc_sequence_value
and nvl(y.TIPO_MANDATO,'X') = 'V'
and x.ANNO_FINANZIARIO = y.ANNO_FINANZIARIO
and x.NUMERO_DISTINTA = y.NUMERO_DISTINTA
and nvl(x.DOCUMENTI_SANITA,'X') = 'S'
order by 1,3;

cursor dettaglio is
select nvl(t.amount,0) importo
,nvl(p.conto_partite_giro,p.codice) codice_conto
,decode(p.codice_beneficiario,'0',p.descrizione||nvl(a.denominazione,a.nome||' '||a.cognome),p.descrizione) descr_conto
from fin_t_documenti d
, fin_t_documenti_terze_parti t
, sa_piano_dei_conti p
,fin_t_anagrafica a
where d.doc_id = id_mand
and t.doc_id = d.doc_id
and a.id = t.id_ana
and (p.codice_beneficiario = t.id_ana OR p.codice_beneficiario = 0)
and p.conto_collegato = conto_coll;

-- cursore per le partite di giro Conto_sanità = 84

BEGIN

 -- inserisco scrittura per debiti vs/aziende a tesoreria
 for i in mand loop -- da togliere e lasciare solo quello attualmente commentato se si decide di ritornare a fare la scritture debiti generici e debiti vs aziende

id_mand := i.doc_id; -- da togliere e lasciare solo quello attualmente commentato se si decide di ritornare a fare la scritture debiti generici e debiti vs aziende
 id_pn_testa := 0;
 conto_coll := nvl(i.conto_partite_giro,i.conto_sanita);
-- trovo la causale contabile relativa al capitolo
  begin
    select nvl(id,0),nvl(denominazione,'X')
      into id_causale,descr_causale
    from sa_causali_contabili
    where capitolo = 'U00000'
    and doc_type = 'MAND';
    exception when no_data_found then id_causale:= 0; descr_causale := '';
  end;
  -- PRIMA NOTA TESTA
  -- stacca id per sa_prima_nota_testa
      select nvl(max(num_reg),0) + 1 into id_pn_testa
      from sa_prima_nota_testa
      where data = i.data_doc;
    --FINE   
    -- inserisco la testata in sa_prima_nota_testa
    insert into sa_prima_nota_testa(DATA,NUM_REG,DOC_ID,DESCRIZIONE_MOV,SA_CC_ID,utente)
    values
    (i.data_doc,
     id_pn_testa,
     i.doc_id,
     i.descrizione,
     id_causale,
     i.utente
    );
    commit;
  -- FINE PNT
  
   for d in dettaglio loop
        -- stacco identificativo univoco e progressivo per il rigo della sa_prima_nota_corpo
        SELECT sq_sapc.NEXTVAL into indice 
        FROM dual;
        -- ****
        -- inserisco scrittura per riduzione conto DEBITI GENERICI e ribaltamento sui conti di Debito delle aziende
        insert into sa_prima_nota_corpo (T_DATA,T_NUM_REG,ID_CONTO,PROGR,D_A,IMPORTO)
        values
        (i.data_doc,
        id_pn_testa,
        d.codice_conto, -- conto DEBITI vs Azienda X
         indice,
         'D',
         nvl(d.importo,0)
        );
        commit;
    end loop;
    
    -- FINE causali guidate
    -- stacco identificativo univoco e progressivo per il rigo della sa_prima_nota_corpo
    SELECT sq_sapc.NEXTVAL into indice 
    FROM dual;
    -- ****
    insert into sa_prima_nota_corpo (T_DATA,T_NUM_REG,ID_CONTO,PROGR,D_A,IMPORTO)
    values
    (i.data_doc,
    id_pn_testa,
    8, -- Tesoreria Sanità
    indice,
    'A',
    nvl(i.importo_mand,0)
    );
    commit;
 
  -- FINE causali guidate
-- FINE scrittura debiti vs/aziende a Tesoreria
  end loop;
