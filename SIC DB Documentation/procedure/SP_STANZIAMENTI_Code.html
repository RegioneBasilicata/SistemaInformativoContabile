+ nvl(b.assestamento_competenza_piu,0)

, to_date('0101'||p_anno,'ddmmyyyy') data_doc

, fin_t_articoli a
where a.anno = p_anno
and a.eu = 'U'
and nvl(a.conto_sanita,'0') <> '0'
and b.anno = a.anno
and b.eu||b.articolo = a.eu||a.codice
