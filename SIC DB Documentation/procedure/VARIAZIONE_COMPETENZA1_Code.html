TEXT 
procedure       variazione_competenza1 
(w_anno in number 
,w_capitolo in varchar2
,w_dal in date
,w_al in date
,w_tipo_atto varchar2
,w_stanziamento_ini out number
,w_variazione_pos out number
,w_variazione_neg out number
,w_stanziamento_fin out number) AS

variazione number(15,2):=0;

begin

 begin
  select 
  nvl(sum(decode(substr(d.doc_type,7,3),'NEG',(d.document_amount * -1),d.document_amount)),0) 
  into variazione
  from  fin_t_documenti d
  where d.segment8 = w_anno
  and d.segment2 = w_capitolo
  and d.doc_type in('V_COM_POS','V_COM_NEG')
  and d.attribute13 = nvl(w_tipo_atto,d.attribute13)
  and trunc(date_created,'dd') between trunc(w_dal,'dd') and trunc(w_al,'dd');
 end;

 if variazione = 0
    then 
    w_variazione_pos := 0;
    w_variazione_neg := 0;
    elsif variazione < 0
          then     
          w_variazione_pos := 0;
          w_variazione_neg := abs(variazione);
          else     
          w_variazione_neg := 0;
          w_variazione_pos := variazione;
 end if;

 begin
  select 
  nvl(sum(decode(substr(d.doc_type,7,3),'NEG',(d.document_amount * -1),d.document_amount)),0) 
  into w_stanziamento_ini
  from  fin_t_documenti d
  where d.segment8 = w_anno
  and d.segment2 = w_capitolo
  and d.doc_type in 
  ('B_COM_POS'
  ,'B_PL1_POS'
  ,'B_PL2_POS'
  ,'V_COM_POS'
  ,'V_PL1_POS'
  ,'V_PL2_POS'
  ,'A_COM_POS'
  ,'A_PL1_POS'
  ,'A_PL2_POS'
  ,'B_COM_NEG'
  ,'B_PL1_NEG'
  ,'B_PL2_NEG'
  ,'V_COM_NEG'
  ,'V_PL1_NEG'
  ,'V_PL2_NEG'
  ,'A_COM_NEG'
  ,'A_PL1_NEG'
  ,'A_PL2_NEG')
  and trunc(date_created,'dd') <trunc(w_dal,'dd');
 end;

 w_stanziamento_fin := w_stanziamento_ini + variazione; 
end;
