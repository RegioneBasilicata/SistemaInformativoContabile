procedure   variazione_competenza_provv
/*
03/02/2019 - Creazione script - serve per gestiore il calcolo delle variazioni prodotte durante l'esercizio provvisorio
*/
(w_anno in number 
,w_capitolo in varchar2
,w_dal in date
,w_al in date
,w_stanziamento_ini out number
,w_variazione_pos out number
,w_variazione_neg out number
,w_stanziamento_fin out number) AS

variazione number(15,2):=0;
variazione_avanzo number(15,2):=0;

begin

-- prendo variazioni
 begin
  select 
  nvl(sum(decode(substr(d.doc_type,7,3),'NEG',(d.document_amount * -1),d.document_amount)),0) 
  into variazione
  from  fin_t_documenti d
  where d.segment8 = w_anno
  and d.segment2 = w_capitolo
  and d.doc_type in('V_COM_POS','V_COM_NEG','A_COM_POS','A_COM_NEG')
  and trunc(date_created,'dd') between trunc(w_dal,'dd') and trunc(w_al,'dd');
 end;


 begin 
   select 
   nvl(sum(decode(substr(d.doc_type,7,3),'NEG',(d.document_amount * -1),d.document_amount)),0) 
   into variazione_avanzo
   from  fin_t_documenti d
   where d.segment8 = w_anno
   and d.segment2 = w_capitolo
   and  d.doc_type in ('B_COM_POS','B_COM_NEG')
   and  nvl(attribute10,'N')||segment9 like 'SAV%'
    and trunc(date_created,'dd') between trunc(w_dal,'dd') and trunc(w_al,'dd'); -- attribute10=S vuol dire utilizzo anticipato e segment9 like AV% vuol dire chiave contabile avanzo Gestione Utilizzo Avanzo Anticipato
 end;

 variazione := nvl(variazione,0)+nvl(variazione_avanzo,0);

 if variazione = 0
    then 
    w_variazione_pos := 0;
    w_variazione_neg := 0;
    elsif variazione < 0
          then     
          w_variazione_pos := 0;
          w_variazione_neg := abs(variazione);
          else     
          w_variazione_neg := 0;
          w_variazione_pos := variazione;
 end if;

 begin
  select 
  nvl(sum(decode(substr(d.doc_type,7,3),'NEG',(d.document_amount * -1),d.document_amount)),0) 
  into w_stanziamento_ini
  from  fin_t_documenti d
  where d.segment8 = w_anno
  and d.segment2 = w_capitolo
  and d.doc_type in 
 ('B_PL1_POS'
 ,'B_PL2_POS'
 ,'V_PL1_POS'
 ,'A_PL1_POS'
 ,'V_PL2_POS'
 ,'A_PL2_POS'
 ,'B_PL1_NEG'
 ,'B_PL2_NEG'
 ,'V_PL1_NEG'
 ,'A_PL1_NEG'
 ,'V_PL2_NEG'
 ,'A_PL2_NEG');
 end;

 w_stanziamento_fin := w_stanziamento_ini + variazione; 
end;
