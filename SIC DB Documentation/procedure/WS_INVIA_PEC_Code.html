procedure ws_invia_pec (pec_serv_ch in varchar2, pec_da in varchar2, pec_a  in varchar2,
                                          pec_oggetto in varchar2, pec_testo in varchar2, pec_id out number) is
    l_envelope CLOB;
    l_risultato xmltype;
    l_mess_id VARCHAR2(4000);
begin
 l_envelope := '<soapenv:Envelope xmlns:soapenv="http://schemas.xmlsoap.org/soap/envelope/" xmlns:ser="http://service.publisys.it/">
   <soapenv:Header/>
   <soapenv:Body>
      <ser:inviaPec>
         <!--Optional:-->
         <fromPec>'||pec_da||'</fromPec>
         <!--Optional:-->
         <utente>infosic@cert.regione.basilicata.it</utente>
         <!--Optional:-->
         <password>INFOSIC12aL</password>
         <!--Optional:-->
         <addressTo>'||pec_a||'</addressTo>
         <!--Optional:-->
         <oggetto>'||pec_oggetto||'</oggetto>
         <!--Optional:-->
         <corpoMessaggio>'||pec_testo||'</corpoMessaggio>
      </ser:inviaPec>
   </soapenv:Body>
</soapenv:Envelope>';
l_risultato:=apex_web_service.make_request(
              p_url => 'http://172.18.17.187:8080/RicezionePecWS/RicezionePec', 
              p_action => 'inviaPec',
              p_envelope => l_envelope);

l_mess_id := APEX_WEB_SERVICE.PARSE_XML(
          p_xml => l_risultato,
          p_xpath => '//return/text()');
l_mess_id:=replace(l_mess_id,'<','<');
l_mess_id:=replace(l_mess_id,'>','>');

select SQ_MAIL_PEC.nextval 
   into pec_id
   from dual;
insert into fnd_mail_pec
(ID, SERVIZIO_CHIAMANTE, MAIL_MITTENTE, MAIL_DESTINATARIO, OGGETTO,
DATA_INVIO, ID_SERVIZIO_MAIL)
values
(pec_id, pec_serv_ch, pec_da, pec_a, pec_oggetto, sysdate, l_mess_id);
commit;
end;
