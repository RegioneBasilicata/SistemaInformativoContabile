procedure ws_parix (cod_fisc in varchar2,risposta out clob) is
  Sec_Id          Number;
	l_envelope CLOB;
	l_risultato xmltype;
	l_ris_string CLOB; --varchar2(4000);
	l_res VARCHAR2(4000);
  l_ns           varchar2(2000);
  l_cod_fisc_tr  varchar2(16);
  l_vers_appl    varchar2(10);
  l_cod_err      varchar2(50);
  l_descr_err    varchar2(2000);
  l_n_rea        varchar2(20);
  l_prov_ccia    varchar2(4);
  doc xmldom.DOMDocument;
  src_clob    BFILE;
  dst_offset  number := 1 ;
  src_offset  number := 1 ;
  lang_ctx    number := DBMS_LOB.DEFAULT_LANG_CTX;
  warning     number;
begin
Select Apex_Util.Find_Security_Group_Id (P_Workspace => 'SIC') 
  Into   Sec_Id
  From   Dual;
Apex_Util.Set_Security_Group_Id(P_Security_Group_Id => Sec_Id);
l_ns:='xmlns:soapenv="http://schemas.xmlsoap.org/soap/envelope/" xmlns:xsd="http://www.w3.org/2001/XMLSchema" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xmlns:ns1="http://ws.icr.ic.it" xmlns:soapenc="http://schemas.xmlsoap.org/soap/encoding/"';

l_cod_fisc_tr:='80002950766';
l_vers_appl:= '1.0';
-- ricerca dati rea da codice fiscale
l_envelope := '<soapenv:Envelope xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xmlns:xsd="http://www.w3.org/2001/XMLSchema" xmlns:soapenv="http://schemas.xmlsoap.org/soap/envelope/" xmlns:ws="http://ws.icr.ic.it">
   <soapenv:Header/>
   <soapenv:Body>
      <ws:RicercaImpresePerCodiceFiscale soapenv:encodingStyle="http://schemas.xmlsoap.org/soap/encoding/">
         <codice_fiscale xsi:type="soapenc:string" xmlns:soapenc="http://schemas.xmlsoap.org/soap/encoding/">00317370773</codice_fiscale>
         <switch_control xsi:type="soapenc:string" xmlns:soapenc="http://schemas.xmlsoap.org/soap/encoding/">no</switch_control>
         <user xsi:type="soapenc:string" xmlns:soapenc="http://schemas.xmlsoap.org/soap/encoding/">WSParixSRA</user>
         <password xsi:type="soapenc:string" xmlns:soapenc="http://schemas.xmlsoap.org/soap/encoding/">JPnGAPk6</password>
      </ws:RicercaImpresePerCodiceFiscale>
   </soapenv:Body>
</soapenv:Envelope>';
l_risultato:=apex_web_service.make_request(
              p_url => 'http://sirb.regione.basilicata.it:8080/parixgate/services/gate', 
              p_action => 'RicercaImpresePerCodiceFiscale',
              p_envelope => l_envelope);

--salvo il risultato in un file in modo da poterlo leggere in un clob e trattarlo leggendo solotanto la <RISPOSTA>
select XMLSERIALIZE(CONTENT (l_risultato)) 
into l_ris_string from dual;
l_ris_string:=replace(l_ris_string,'<','<');
l_ris_string:=replace(l_ris_string,'>','>');
l_ris_string:=replace(l_ris_string,'"','"');
l_ris_string:=substr(l_ris_string,instr(l_ris_string,'<RISPOSTA>'),instr(l_ris_string,'</RISPOSTA>')-instr(l_ris_string,'<RISPOSTA>')+11);
l_risultato:=xmltype(l_ris_string);
l_res := APEX_WEB_SERVICE.PARSE_XML(
         p_xml => l_risultato,
         p_xpath => '//RISPOSTA/HEADER/ESITO/text()');

--risposta:=l_risultato.getclobval(); --l_res; -- canc         

if l_res='OK' then
  l_n_rea:= APEX_WEB_SERVICE.PARSE_XML(
            p_xml => l_risultato,
            p_xpath => '//RISPOSTA/DATI/LISTA_IMPRESE/ESTREMI_IMPRESA/DATI_ISCRIZIONE_REA/NREA/text()');
  l_prov_ccia := APEX_WEB_SERVICE.PARSE_XML(
            p_xml => l_risultato,
            p_xpath => '//RISPOSTA/DATI/LISTA_IMPRESE/ESTREMI_IMPRESA/DATI_ISCRIZIONE_REA/CCIAA/text()');
-- Letura dati impresa            
  l_envelope := '<soapenv:Envelope xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xmlns:xsd="http://www.w3.org/2001/XMLSchema" xmlns:soapenv="http://schemas.xmlsoap.org/soap/envelope/" xmlns:ws="http://ws.icr.ic.it">
     <soapenv:Header/>
     <soapenv:Body>
         <ws:DettaglioCompletoImpresa soapenv:encodingStyle="http://schemas.xmlsoap.org/soap/encoding/">
           <sgl_prv_sede xsi:type="soapenc:string" xmlns:soapenc="http://schemas.xmlsoap.org/soap/encoding/">'||l_prov_ccia||'</sgl_prv_sede>
           <n_rea_sede xsi:type="soapenc:string" xmlns:soapenc="http://schemas.xmlsoap.org/soap/encoding/">'||l_n_rea||'</n_rea_sede>
           <switch_control xsi:type="soapenc:string" xmlns:soapenc="http://schemas.xmlsoap.org/soap/encoding/"></switch_control>
           <abilitaSoci xsi:type="soapenc:string" xmlns:soapenc="http://schemas.xmlsoap.org/soap/encoding/">si</abilitaSoci>
           <user xsi:type="soapenc:string" xmlns:soapenc="http://schemas.xmlsoap.org/soap/encoding/">WSParixSRA</user>
           <password xsi:type="soapenc:string" xmlns:soapenc="http://schemas.xmlsoap.org/soap/encoding/">JPnGAPk6</password>
        </ws:DettaglioCompletoImpresa>
     </soapenv:Body>
  </soapenv:Envelope>';
  l_risultato:=apex_web_service.make_request(
                p_url => 'http://sirb.regione.basilicata.it:8080/parixgate/services/gate', 
                p_action => 'DettaglioCompletoImpresa',
                p_envelope => l_envelope);

  doc := xmldom.newDOMDocument(l_risultato);
  xmldom.writeToFile(doc, 'DIR_PROVVEDIMENTI/parix'||l_n_rea||'.xml');
  xmldom.freeDocument(doc);
  
  src_clob := BFILENAME('DIR_PROVVEDIMENTI', 'parix'||l_n_rea||'.xml');
  DBMS_LOB.fileopen(src_clob, DBMS_LOB.file_readonly);
  dbms_lob.createtemporary(l_ris_string, true);

  DBMS_LOB.LoadCLOBFromFile(
          DEST_LOB     => l_ris_string
        , SRC_BFILE    => src_clob
        , AMOUNT       => DBMS_LOB.GETLENGTH(src_clob)
        , DEST_OFFSET  => dst_offset
        , SRC_OFFSET   => src_offset
        , BFILE_CSID   => DBMS_LOB.DEFAULT_CSID
        , LANG_CONTEXT => lang_ctx
        , WARNING      => warning
    );

  DBMS_LOB.CLOSE(src_clob);

  l_ris_string:=replace(l_ris_string,'<','<');
  l_ris_string:=replace(l_ris_string,'>','>');
  l_ris_string:=replace(l_ris_string,'"','"');
  l_ris_string:=substr(l_ris_string,instr(l_ris_string,'<RISPOSTA>'),instr(l_ris_string,'</RISPOSTA>')-instr(l_ris_string,'<RISPOSTA>')+11);
  doc := xmldom.newDOMDocument(l_ris_string);
  xmldom.writeToFile(doc, 'DIR_PROVVEDIMENTI/parix'||l_n_rea||'.xml');
  xmldom.freeDocument(doc);
  DBMS_LOB.FREETEMPORARY(l_ris_string);  
  risposta:=l_res;

/*

end;
