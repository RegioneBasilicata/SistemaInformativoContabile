procedure ws_verifica_pec (pec_id in number,pec_risp out varchar2) is
    l_envelope CLOB;
    l_risultato xmltype;
    l_mess_id VARCHAR2(4000);
begin
 select id_servizio_mail
  into l_mess_id
  from fnd_mail_pec
  where id=pec_id;
 l_envelope := '<soapenv:Envelope xmlns:soapenv="http://schemas.xmlsoap.org/soap/envelope/" xmlns:ser="http://service.publisys.it/">
   <soapenv:Header/>
   <soapenv:Body>
      <ser:confermaPec>
         <!--Optional:-->
         <utente>infosic@cert.regione.basilicata.it</utente>
         <!--Optional:-->
         <password>INFOSIC12aL</password>
         <!--Optional:-->
         <messageID>'||l_mess_id||'</messageID>
      </ser:confermaPec>
   </soapenv:Body>
</soapenv:Envelope>';
l_risultato:=apex_web_service.make_request(
              p_url => 'http://172.18.17.187:8080/RicezionePecWS/RicezionePec', 
              p_action => 'confermaPec',
              p_envelope => l_envelope);

-- controllo risposta indirizzo pec
pec_risp := apex_web_service.parse_xml(p_xml => l_risultato, p_xpath => '//return[2]/oggetto/text()');
if pec_risp is null then
-- controllo risposta indirizzo email normale
   pec_risp := apex_web_service.parse_xml(p_xml => l_risultato, p_xpath => '//return[1]/oggetto/text()');
end if;
if pec_risp is null then
-- controllo risposta indirizzo email normale
   pec_risp := 'NESSUNA RISPOSTA';
end if;


update fnd_mail_pec
 set risposta=pec_risp
 where ID=pec_id;
commit;
end;
