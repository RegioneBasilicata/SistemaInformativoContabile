TEXT 
trigger aggiorna_fnd_reports_categorie
	AFTER INSERT OR DELETE OR UPDATE ON fnd_reports_categorie
	FOR EACH ROW
	DECLARE

		wcodice NUMBER;
		wdescrizione VARCHAR2(2000);
		wcodice_padre NUMBER;

	BEGIN

		IF INSERTING OR UPDATING THEN

			wcodice := :new.codice;
			wdescrizione := :new.descrizione;
			wcodice_padre := :new.codice_padre;

			IF UPDATING THEN

				BEGIN

					delete from SIC_ALSIA.fnd_reports_categorie where codice = wcodice;
					delete from SIC_APT.fnd_reports_categorie where codice = wcodice;
					delete from SIC_ARBEA.fnd_reports_categorie where codice = wcodice;
					delete from SIC_ARDSU.fnd_reports_categorie where codice = wcodice;
					delete from SIC_ARPAB.fnd_reports_categorie where codice = wcodice;
					delete from SIC_CONSIGLIO.fnd_reports_categorie where codice = wcodice;
					delete from SIC_PARCOGALLIPOLI.fnd_reports_categorie where codice = wcodice;
					delete from SIC_PARCOMURGIA.fnd_reports_categorie where codice = wcodice;
					delete from SIC_LAB.fnd_reports_categorie where codice = wcodice;

				END;

			END IF;

			BEGIN

				insert into SIC_ALSIA.fnd_reports_categorie (codice, descrizione, codice_padre) values (wcodice, wdescrizione, wcodice_padre);
				insert into SIC_APT.fnd_reports_categorie (codice, descrizione, codice_padre) values (wcodice, wdescrizione, wcodice_padre);
				insert into SIC_ARBEA.fnd_reports_categorie (codice, descrizione, codice_padre) values (wcodice, wdescrizione, wcodice_padre);
				insert into SIC_ARDSU.fnd_reports_categorie (codice, descrizione, codice_padre) values (wcodice, wdescrizione, wcodice_padre);
				insert into SIC_ARPAB.fnd_reports_categorie (codice, descrizione, codice_padre) values (wcodice, wdescrizione, wcodice_padre);
				insert into SIC_CONSIGLIO.fnd_reports_categorie (codice, descrizione, codice_padre) values (wcodice, wdescrizione, wcodice_padre);
				insert into SIC_PARCOGALLIPOLI.fnd_reports_categorie (codice, descrizione, codice_padre) values (wcodice, wdescrizione, wcodice_padre);
				insert into SIC_PARCOMURGIA.fnd_reports_categorie (codice, descrizione, codice_padre) values (wcodice, wdescrizione, wcodice_padre);
				insert into SIC_LAB.fnd_reports_categorie (codice, descrizione, codice_padre) values (wcodice, wdescrizione, wcodice_padre);

			END;

		ELSE

			wcodice := :old.codice;

			BEGIN

					delete from SIC_ALSIA.fnd_reports_categorie where codice = wcodice;
					delete from SIC_APT.fnd_reports_categorie where codice = wcodice;
					delete from SIC_ARBEA.fnd_reports_categorie where codice = wcodice;
					delete from SIC_ARDSU.fnd_reports_categorie where codice = wcodice;
					delete from SIC_ARPAB.fnd_reports_categorie where codice = wcodice;
					delete from SIC_CONSIGLIO.fnd_reports_categorie where codice = wcodice;
					delete from SIC_PARCOGALLIPOLI.fnd_reports_categorie where codice = wcodice;
					delete from SIC_PARCOMURGIA.fnd_reports_categorie where codice = wcodice;
					delete from SIC_LAB.fnd_reports_categorie where codice = wcodice;

			END;

		END IF;

	END aggiorna_fnd_reports_categorie;
