TRIGGER duplic_dir_validatori
AFTER INSERT OR UPDATE ON fin_t_cdg_dirigenti
DECLARE 
 v VARCHAR2(20);
 CURSOR duplicazioni_rich IS
  SELECT ufficio
  FROM fin_t_cdg_dirigenti
  WHERE NVL(fl_richcondiv,'N') = 'S'
  GROUP BY ufficio
  HAVING COUNT(*) > 1;
 CURSOR duplicazioni_mon IS
  SELECT ufficio
  FROM fin_t_cdg_dirigenti
  WHERE NVL(fl_schmon,'N') = 'S'
  GROUP BY ufficio
  HAVING COUNT(*) > 1;
BEGIN 
 OPEN duplicazioni_rich;
 FETCH duplicazioni_rich
 INTO v;
 IF duplicazioni_rich%FOUND
  THEN
   CLOSE duplicazioni_rich;
   RAISE_APPLICATION_ERROR(20001,'Salvataggio non consentito: per almeno un ufficio, si sta cercando di indicare piu'' di un Dirigente che valida le richieste di condivisione!');
  ELSE
   CLOSE duplicazioni_rich;
 END IF;
 OPEN duplicazioni_mon;
 FETCH duplicazioni_mon
 INTO v;
 IF duplicazioni_mon%FOUND
  THEN
   CLOSE duplicazioni_mon;
   RAISE_APPLICATION_ERROR(20001,'Salvataggio non consentito: per almeno un ufficio, si sta cercando di indicare piu'' di un Dirigente che valida le schede di monitoraggio!');
  ELSE
   CLOSE duplicazioni_mon;
 END IF;
END;
