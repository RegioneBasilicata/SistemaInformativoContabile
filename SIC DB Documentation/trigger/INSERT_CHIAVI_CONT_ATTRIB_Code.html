TRIGGER INSERT_CHIAVI_CONT_ATTRIB
BEFORE INSERT ON fin_t_chiavi_contabili_attrib
FOR EACH ROW
DECLARE
  a_progr number;
BEGIN
  SELECT max(id)+1
    INTO a_progr
    FROM fin_t_chiavi_contabili_attrib;
  :new.id := nvl(a_progr,1);
END;
