TEXT 
TRIGGER INSERT_FATT_IMP
BEFORE INSERT ON FIN_T_FATT_IMPORT
FOR EACH ROW
DECLARE
  a_progr number;
  a_tab_codice number;
  a_gen_progr number;
  a_fornitore number(10);
  a_tipo_pag  number(5);
  a_iban varchar2(27);
BEGIN
  SELECT SEQ_FAT_IMP.nextval
    INTO a_progr
    FROM dual;
  :new.progr := a_progr;
  :new.UTENTE := nvl(v('APP_USER'),user);
  if :new.GEN_PROGR is null then
     select max(TAB_CODICE),max(PROGR)
       into a_tab_codice,a_progr
       from bi_generale_gest
       where col_rif_file=:new.POD;
     :new.GEN_PROGR:=a_progr;
     :new.TAB_CODICE:=a_tab_codice;
  end if;
  if :new.FORNITORE is null then
     select max(fornitore)
       into a_fornitore
       from bi_fornitori_contratti
       where TAB_CODICE=:new.tab_codice and
             GEN_PROGR=:new.gen_progr;
     :new.fornitore:=a_fornitore;
  end if;
  if :new.fornitore is not null and
     (:new.TIPO_PAGAM is null or
     :new.IBAN is null) then
     select max(iban),max(pagam_default)
       into a_iban,a_tipo_pag
       from bi_fornitori_contratti
       where TAB_CODICE=:new.tab_codice and
             GEN_PROGR=:new.gen_progr and
             FORNITORE=:new.fornitore;
     if :new.TIPO_PAGAM is null then
        :new.TIPO_PAGAM:=a_tipo_pag;
     end if;   
     if :new.IBAN is null then
        :new.IBAN:=a_iban;
     end if;   
  end if;
END;
