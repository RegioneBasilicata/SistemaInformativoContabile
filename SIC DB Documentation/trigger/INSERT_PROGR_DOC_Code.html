TRIGGER INSERT_PROGR_DOC
BEFORE INSERT ON fin_t_doc_fat
FOR EACH ROW
DECLARE
  a_progr number;
BEGIN
  SELECT SEQ_DOC_FAT.nextval
    INTO a_progr
    FROM dual;
  :new.progr := a_progr;
  :new.UTENTE := nvl(v('APP_USER'),user);
END;
