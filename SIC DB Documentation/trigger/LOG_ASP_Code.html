TRIGGER "GAT2".LOG_ASP
AFTER DELETE OR UPDATE
ON FIN_T_ASP
FOR EACH ROW



declare
tipo_aggiornamento varchar2(1);
begin
if updating
   then tipo_aggiornamento := 'U';
   else tipo_aggiornamento := 'D';
end if;
insert into fin_t_log_asp values
(:OLD.ANNO,
:OLD.CODICE,
:OLD.DESCRIZIONE,
:OLD.DATA_ASSESTAMENTO,
SYSDATE,
tipo_aggiornamento,
:OLD.aggiornato_da);
end;
