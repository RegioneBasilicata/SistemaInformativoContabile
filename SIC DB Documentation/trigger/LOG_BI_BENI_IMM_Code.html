TEXT 
TRIGGER "GAT2".LOG_BI_BENI_IMM
AFTER DELETE OR UPDATE
ON BI_BENI_IMM
FOR EACH ROW

declare
tipo_aggiornamento varchar2(1);
begin
if updating
   then tipo_aggiornamento := 'U';
   else tipo_aggiornamento := 'D';
end if;
insert into BI_LOG_BENI_IMM values
(:OLD.CODICE,
  :OLD.DESCRIZIONE,
  :OLD.TIPO_BENE,
  :OLD.STOR_ART,
  :OLD.DATA_ISCR,
  :OLD.VAL_MERC,
  :OLD.VAL_FISC,
  :OLD.NOTE,
  :OLD.CREATO_DA,
  :OLD.CREATO_IL,
  :OLD.MODIFICATO_DA,
  SYSDATE,
  tipo_aggiornamento);
end;
