ON BI_CATASTO_FAB

  :OLD.SITUAZIONE_AL ,

  :OLD.CONSISTENZA   ,

  :OLD.QUOTA_POSS,

  :OLD.REND_CAT      ,
  :OLD.REND_FISC     ,
  :OLD.CREATO_DA     ,
  :OLD.CREATO_IL     ,
  :OLD.MODIFICATO_DA ,
  SYSDATE ,
  tipo_aggiornamento);
end;
