TEXT 
TRIGGER "GAT2".LOG_BI_CATASTO_TER
AFTER DELETE OR UPDATE
ON BI_CATASTO_TER
FOR EACH ROW

declare
tipo_aggiornamento varchar2(1);
begin
if updating
   then tipo_aggiornamento := 'U';
   else tipo_aggiornamento := 'D';
end if;
insert into BI_LOG_CATASTO_TER values
( :OLD.COD_CESP      ,
  :OLD.SITUAZIONE_AL ,
  :OLD.COMUNE        ,
  :OLD.FOGLIO,
  :OLD.PARTICELLA,
  :OLD.CLASSE        ,
  :OLD.SUPERFICIE    ,
  :OLD.QUALITA       ,
  :OLD.REDD_DOMIN    ,
  :OLD.REDD_AGR     ,
  :OLD.TIPO_PROPR,
  :OLD.QUOTA_POSS,
  :OLD.NOTE,
  :OLD.CREATO_DA     ,
  :OLD.CREATO_IL     ,
  :OLD.MODIFICATO_DA ,
  SYSDATE,
  tipo_aggiornamento);
end;
