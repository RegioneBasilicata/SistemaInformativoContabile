TEXT 
TRIGGER "GAT2".LOG_BI_DISMISSIONI
AFTER DELETE OR UPDATE
ON BI_DISMISSIONI
FOR EACH ROW

declare
tipo_aggiornamento varchar2(1);
begin
if updating
   then tipo_aggiornamento := 'U';
   else tipo_aggiornamento := 'D';
end if;
insert into BI_LOG_DISMISSIONI values
( :OLD.COD_CESP       ,
  :OLD.TIPO_DISM      ,
  :OLD.DATA_DISM      ,
  :OLD.IMP_VEND       ,
  :OLD.NOTE_DIS       ,
  :OLD.CONTRAENTI     ,
  :OLD.TIPO_ATTO      ,
  :OLD.DATA_ATTO      ,
  :OLD.NUM_ATTO       ,
  :OLD.NOTE_ATTO      ,
  :OLD.CREATO_DA     ,
  :OLD.CREATO_IL     ,
  :OLD.MODIFICATO_DA ,
  SYSDATE,
  tipo_aggiornamento);
end;
