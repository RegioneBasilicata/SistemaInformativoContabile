TEXT 
TRIGGER "GAT2".LOG_BI_PROVENIENZE
AFTER DELETE OR UPDATE
ON BI_PROVENIENZE
FOR EACH ROW

declare
tipo_aggiornamento varchar2(1);
begin
if updating
   then tipo_aggiornamento := 'U';
   else tipo_aggiornamento := 'D';
end if;
insert into BI_LOG_PROVENIENZE values
(  :OLD.COD_CESP       ,
  :OLD.PROGR          ,
  :OLD.TIPO_PROV      ,
  :OLD.TIPO_ATTO      ,
  :OLD.DATA_ATTO      ,
  :OLD.NUM_ATTO       ,
  :OLD.NOTE           ,
  :OLD.LUOGO_TRASCR,
  :OLD.DATA_TRASCR,
  :OLD.NUM_REG_GEN,
  :OLD.NUM_REG_PART,
  :OLD.CREATO_DA     ,
  :OLD.CREATO_IL     ,
  :OLD.MODIFICATO_DA ,
  SYSDATE,
  tipo_aggiornamento);
end;
