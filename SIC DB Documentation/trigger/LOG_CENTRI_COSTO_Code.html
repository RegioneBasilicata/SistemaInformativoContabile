TRIGGER LOG_CENTRI_COSTO
  before update on ap_centri_costo           
  for each row 
declare
nominativo varchar2(2000);
begin  
select max(cognome||' '||nome)
 into nominativo
 from fnd_user
 where user_name = nvl(v('APP_USER'),user);

insert into ap_centri_costo_log
 (codice,descrizione,da_periodo,a_periodo,imp_max,km_max,blocco_sn,fondo_sn,tipo_spese,data_mod,utente_mod,nominativo_mod,COSTO_KM)
 values
 (:old.codice,:old.descrizione,:old.da_periodo,:old.a_periodo,:old.imp_max,:old.km_max,:old.BLOCCO_SN,:OLD.fondo_sn,:OLD.tipo_spese,sysdate,nvl(v('APP_USER'),user),nominativo,:OLD.COSTO_KM);

end;
