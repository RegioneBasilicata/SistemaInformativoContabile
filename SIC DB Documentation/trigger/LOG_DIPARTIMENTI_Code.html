TEXT 
TRIGGER "GAT2".LOG_DIPARTIMENTI
AFTER DELETE OR UPDATE
ON FIN_T_DIPARTIMENTI
FOR EACH ROW

declare
tipo_aggiornamento varchar2(1);
begin
if updating
   then tipo_aggiornamento := 'U';
   else tipo_aggiornamento := 'D';
end if;
insert into fin_t_log_dipartimenti values
( :OLD.ANNO,
  :OLD.CODICE,
  :OLD.DESCRIZIONE,
  :OLD.ATTIVO,
  :OLD.NOTE,
  :OLD.DATA_ASSESTAMENTO,
SYSDATE,
tipo_aggiornamento,
:OLD.aggiornato_da);
end;
