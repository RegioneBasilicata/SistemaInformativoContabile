TEXT 
TRIGGER "GAT2".LOG_EC_movimenti
AFTER DELETE OR UPDATE
ON EC_MOVIMENTI
FOR EACH ROW
declare
tipo_aggiornamento varchar2(1);
begin
if updating
   then tipo_aggiornamento := 'U';
   else tipo_aggiornamento := 'D';
end if;
insert into ec_log_movimenti values
(
  :OLD.TIPO_MOV     ,
  :OLD.DATA         ,
  :OLD.NUMERO       ,
  :OLD.UFFICIO      ,
  :OLD.FORNITORE    ,
  :OLD.SEDE         ,
  :OLD.PROG_RICH    ,
  :OLD.DATA_DOC     ,
  :OLD.NUM_DOC      ,
  :OLD.NOTE         ,
  :OLD.FLAG         ,
  :OLD.FORN_STR     ,
  :OLD.MOTIVO       ,
  :OLD.CAP_SPESA    ,
  :OLD.ATTO_AUTOR   ,
  :OLD.MOD_CONS     ,
  :OLD.CREATO_DA    ,
  :OLD.CREATO_IL    ,
  :OLD.MODIFICATO_DA,
  sysdate,
  tipo_aggiornamento
  );
end;
