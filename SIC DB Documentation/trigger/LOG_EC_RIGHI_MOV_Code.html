TEXT 
TRIGGER "GAT2".LOG_EC_RIGHI_MOV
AFTER DELETE OR UPDATE
ON EC_RIGHI_MOV
FOR EACH ROW
declare
tipo_aggiornamento varchar2(1);
begin
if updating
   then tipo_aggiornamento := 'U';
   else tipo_aggiornamento := 'D';
end if;
insert into ec_log_RIGHI_MOV
(UFFICIO,TIPO_MOV,DATA,NUMERO,PROGR,COD_ART,
QTA,DESCRIZIONE,SCONTO,PREZZO,QTA_EVASA,DATA_RICH,
NUM_RICH,EVASO,TIPO_MODIFICA,DATA_MODIFICA) values
(
  :OLD.UFFICIO      ,
  :OLD.TIPO_MOV     ,
  :OLD.DATA         ,
  :OLD.NUMERO       ,
  :OLD.PROGR        ,
  :OLD.COD_ART      ,
  :OLD.QTA          ,
  :OLD.DESCRIZIONE  ,
  :OLD.SCONTO       ,
  :OLD.PREZZO       ,
  :OLD.QTA_EVASA    ,
  :OLD.DATA_RICH    ,
  :OLD.NUM_RICH     ,
  :OLD.EVASO        ,
  tipo_aggiornamento,
  sysdate
);
end;
