TEXT 
TRIGGER "GAT2".LOG_OB_GEST 
AFTER DELETE OR UPDATE
ON FIN_T_OBIETTIVI_GESTIONALI
FOR EACH ROW



declare
tipo_aggiornamento varchar2(1);
begin
if updating
   then tipo_aggiornamento := 'U';
   else tipo_aggiornamento := 'D';
end if;
insert into fin_t_log_ob_gest values
(:OLD.ANNO,
:OLD.CODICE_PARTE1,
:OLD.CODICE_PARTE2,
:OLD.CODICE_PARTE3,
:OLD.FIN_T_ASP_ANNO,
:OLD.FIN_T_ASP_CODICE,
:OLD.FIN_T_DIRETTRICE_ANNO,
:OLD.FIN_T_DIRETTRICE_CODICE,
:OLD.FIN_T_RISULTATI_ANNO,
:OLD.FIN_T_RISULTATI_CODICE,
:OLD.DESCRIZIONE,
:OLD.VALIDO_DAL,
:OLD.VALIDO_AL,
:OLD.DATA_ASSESTAMENTO,
SYSDATE,
tipo_aggiornamento,
:OLD.aggiornato_da,
:OLD.peso);
end;
