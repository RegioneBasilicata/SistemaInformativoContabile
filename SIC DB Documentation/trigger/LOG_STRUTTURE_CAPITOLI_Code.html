TEXT 
TRIGGER "GAT2".LOG_STRUTTURE_CAPITOLI AFTER
  DELETE OR
  UPDATE
    ON FIN_T_STRUTTURE_CAPITOLI FOR EACH ROW DECLARE tipo_aggiornamento
    VARCHAR2(1);
  BEGIN
    IF updating THEN
      tipo_aggiornamento := 'U';
    ELSE
      tipo_aggiornamento := 'D';
    END IF;
    INSERT
    INTO
      fin_t_log_strutture_capitoli VALUES
      (
        :OLD.FIN_T_DIPARTIMENTI_ANNO,
        :OLD.FIN_T_DIPARTIMENTI_CODICE,
        :OLD.FIN_T_STRUTTURE_ANNO,
        :OLD.FIN_T_STRUTTURE_CODICE,
        :OLD.CAPITOLO_ANNO,
        :OLD.CAPITOLO_EU,
        :OLD.CAPITOLO_CODICE,
        :OLD.VALIDITA_DAL,
        :OLD.VALIDITA_AL,
        :OLD.DATA_ASSESTAMENTO,
        SYSDATE,
        tipo_aggiornamento,
        :OLD.aggiornato_da,
        :OLD.titolarita,
        :OLD.importo_condivisione,
        :OLD.data_i_condivisione,
        :OLD.data_f_condivisione,
        :OLD.importo_impegnato,
        :OLD.importo_prenotato
      );
  END;
