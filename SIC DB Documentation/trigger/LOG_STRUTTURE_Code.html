TEXT 
TRIGGER "GAT2".LOG_STRUTTURE
AFTER DELETE OR UPDATE
ON FIN_T_STRUTTURE
FOR EACH ROW


declare
tipo_aggiornamento varchar2(1);
begin
if updating
   then tipo_aggiornamento := 'U';
   else tipo_aggiornamento := 'D';
end if;
insert into fin_t_log_strutture values
( :OLD.ANNO ,
  :OLD.CODICE,
  :OLD.FIN_T_DIPARTIMENTI_ANNO,
  :OLD.FIN_T_DIPARTIMENTI_CODICE,
  :OLD.DESCRIZIONE,
  :OLD.ATTIVO,
  :OLD.NOTE,
  :OLD.DATA_ASSESTAMENTO,
  :OLD.MONITORAGGIO,
SYSDATE,
tipo_aggiornamento,
:OLD.aggiornato_da);
end;
