TEXT 
TRIGGER PCCAF_TR 
before insert or update on PCC_AZIONI_FATT
for each row
begin
    if inserting then
        :new.creato_il := sysdate;
        :new.creato_da := apex_procedure.getIdUtente(nvl(v('APP_USER'),user));
    elsif updating then
        :new.aggiornato_il := sysdate;
        :new.aggiornato_da := apex_procedure.getIdUtente(nvl(v('APP_USER'),user));
    end if;
end;
