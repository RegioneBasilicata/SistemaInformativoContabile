TRIGGER pdp_personale_prestato
 /* Se la risorsa prestata si trova nella tabella PDP_AZIONI_PERSONE 
    con lo stesso anno, non faccio fare nè update nè delete */
BEFORE DELETE OR UPDATE OF matricola ON pdp_personale_prestato
FOR EACH ROW
DECLARE
 v_p_ufficio VARCHAR2(20);
BEGIN
 IF DELETING OR
    :old.matricola != NVL(:new.matricola,:old.matricola+1)
  THEN
   SELECT p_ufficio
   INTO v_p_ufficio
   FROM pdp_azioni_persone
   WHERE matricola = :old.matricola AND
         uff_app_previsione = :old.ufficio_appartenenza AND
         anno = :old.anno AND
         ROWNUM = 1;
   RAISE_APPLICATION_ERROR(-20000,
         'Operazione non consentita: la matricola '|| :old.matricola||' per l''anno '||:old.anno||
         ' risulta impiegata su azioni dell''ufficio '||v_p_ufficio);
 END IF;
EXCEPTION
  WHEN NO_DATA_FOUND 
    THEN NULL;
END;
