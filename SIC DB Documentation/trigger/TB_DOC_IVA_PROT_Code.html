TRIGGER TB_DOC_IVA_PROT
  before insert on TB_DOC_IVA              
  for each row 
begin  
  if :new.TBTD_SIGLA in ('FF','NR','NT') and
    :new.NUM_REG is null then
    select nvl(max(NUM_REG),0)+1
      into :new.NUM_REG
      from TB_DOC_IVA;
  end if;
end;
