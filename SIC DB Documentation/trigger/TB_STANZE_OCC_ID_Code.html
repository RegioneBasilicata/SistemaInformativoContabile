TRIGGER TB_STANZE_OCC_ID  
  before insert on TB_STANZE_OCC              
  for each row 
begin 
 if :new.ID is null then
  for c1 in ( 
    select SEQ_TBSO.nextval next_val 
    from dual 
  ) loop 
    :new.ID :=  c1.next_val; 
  end loop; 
 end if; 
 :new.CREATO_DA := nvl(v('APP_USER'),user);
 :new.CREATO_IL := SYSDATE;
end;
