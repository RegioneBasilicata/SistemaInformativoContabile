TEXT 
TRIGGER TGCM_TRG
before insert or update on TG_CASSEEC_MOV
for each row
begin
    if inserting then
        :new.creato_il := sysdate;
        :new.creato_da := apex_procedure.getIdUtente(nvl(v('APP_USER'),user));
    elsif updating then
        :new.modificato_il := sysdate;
        :new.modificato_da := apex_procedure.getIdUtente(nvl(v('APP_USER'),user));
    end if;
end;
