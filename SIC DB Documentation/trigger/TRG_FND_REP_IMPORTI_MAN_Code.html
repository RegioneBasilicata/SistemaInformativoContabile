TRIGGER trg_fnd_rep_importi_man
BEFORE INSERT OR UPDATE ON fnd_reports_importi_man
FOR EACH ROW 
BEGIN
 IF INSERTING
  THEN
   :NEW.utente_creazione := UPPER(NVL(v('APP_USER'),USER));
   :NEW.data_creazione := SYSDATE;
 ELSIF UPDATING
  THEN
   :NEW.utente_aggiornamento := UPPER(NVL(v('APP_USER'),USER));
   :NEW.data_aggiornamento := SYSDATE;
 END IF; 
END;
