TRIGGER trg_mon_uff_proroghe
BEFORE INSERT OR UPDATE ON mon_uffici_proroghe 
FOR EACH ROW 
BEGIN
 IF INSERTING
  THEN
   :NEW.utente_creazione := UPPER(NVL(v('APP_USER'),'NESSUN UTENTE APEX'));
   :NEW.data_creazione := SYSDATE;
 ELSIF UPDATING
  THEN
   :NEW.utente_aggiornamento := UPPER(NVL(v('APP_USER'),'NESSUN UTENTE APEX'));
   :NEW.data_aggiornamento := SYSDATE;
 END IF; 
END;
