TRIGGER TRIGGER_PIANO_CONTI 
BEFORE INSERT OR UPDATE ON FIN_T_ARTICOLI 
FOR EACH ROW 
BEGIN
 if (:NEW.piano_dei_conti is null) 
   then 
    if :NEW.eu = 'E'
    then :NEW.piano_dei_conti := 'R0001005';
    else :NEW.piano_dei_conti := 'C0303GBL';
   end if;
 end if;
 
END;
